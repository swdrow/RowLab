```typescript
import { motion } from 'framer-motion';
import { Calendar, Users, MapPin, Clock } from 'lucide-react';
import { useHostAthleteVisits } from '@v2/hooks/useRecruitVisits';
import useAuthStore from '@/store/authStore';
import { formatDistanceToNow } from 'date-fns';
import { Link } from 'react-router-dom';

interface HostVisitsWidgetProps {
  className?: string;
}

export function HostVisitsWidget({ className }: HostVisitsWidgetProps) {
  const { user } = useAuthStore();
  // Assuming user has athleteId or we can derive it
  const athleteId = user?.athleteId;

  const { data, isLoading } = useHostAthleteVisits(athleteId || '');

  // Don't render for non-athletes or coaches without athlete profile
  if (!athleteId) return null;

  const visits = data?.visits || [];
  const upcomingVisits = visits.filter(v => v.status === 'scheduled');

  if (upcomingVisits.length === 0) return null;

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      className={`bg-surface-elevated rounded-xl border border-bdr p-4 ${className || ''}`}
    >
      <div className="flex items-center gap-2 mb-4">
        <Users className="w-5 h-5 text-violet-500" />
        <h3 className="font-semibold text-txt-primary">Your Upcoming Hosting Duties</h3>
      </div>

      <div className="space-y-3">
        {upcomingVisits.slice(0, 3).map(visit => (
          <Link
            key={visit.id}
            to={`/app/recruiting/${visit.id}`}
            className="block p-3 bg-surface rounded-lg border border-bdr hover:border-violet-500/50 transition"
          >
            <div className="flex items-start justify-between">
              <div>
                <p className="font-medium text-txt-primary">{visit.recruitName}</p>
                <div className="flex items-center gap-2 text-sm text-txt-secondary mt-1">
                  <Calendar className="w-3.5 h-3.5" />
                  <span>{new Date(visit.date).toLocaleDateString()}</span>
                  <Clock className="w-3.5 h-3.5 ml-2" />
                  <span>{visit.startTime} - {visit.endTime}</span>
                </div>
              </div>
              <span className="text-xs text-violet-500 bg-violet-500/10 px-2 py-1 rounded">
                {formatDistanceToNow(new Date(visit.date), { addSuffix: true })}
              </span>
            </div>
          </Link>
        ))}
      </div>

      {upcomingVisits.length > 3 && (
        <Link
          to="/app/recruiting"
          className="block text-center text-sm text-violet-500 hover:underline mt-3"
        >
          View all {upcomingVisits.length} visits
        </Link>
      )}
    </motion.div>
  );
}
```

```typescript
import { motion, AnimatePresence } from 'framer-motion';
import { X, Calendar, Clock, User, Mail, Phone, School, Share2, FileText, Edit, Trash2 } from 'lucide-react';
import { useRecruitVisit, useDeleteRecruitVisit, useGenerateShareToken } from '@v2/hooks/useRecruitVisits';
import { RichTextDisplay } from '@v2/components/common';
import { useState } from 'react';

interface VisitDetailPanelProps {
  visitId: string | null;
  onClose: () => void;
  onEdit?: (visitId: string) => void;
}

export function VisitDetailPanel({ visitId, onClose, onEdit }: VisitDetailPanelProps) {
  const { data, isLoading } = useRecruitVisit(visitId || '');
  const deleteMutation = useDeleteRecruitVisit();
  const shareTokenMutation = useGenerateShareToken(visitId || '');
  const [showDeleteConfirm, setShowDeleteConfirm] = useState(false);

  const visit = data?.visit;

  const handleDelete = async () => {
    if (!visitId) return;
    await deleteMutation.mutateAsync(visitId);
    onClose();
  };

  const handleShare = async () => {
    if (!visitId) return;
    const result = await shareTokenMutation.mutateAsync();
    // Copy share link to clipboard
    const shareUrl = `${window.location.origin}/visit/${result.shareToken}`;
    navigator.clipboard.writeText(shareUrl);
    // Show toast
  };

  return (
    <AnimatePresence>
      {visitId && (
        <>
          {/* Backdrop */}
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            onClick={onClose}
            className="fixed inset-0 bg-black/50 z-40"
          />

          {/* Panel */}
          <motion.div
            initial={{ x: '100%' }}
            animate={{ x: 0 }}
            exit={{ x: '100%' }}
            transition={{ type: 'spring', damping: 25, stiffness: 300 }}
            className="fixed right-0 top-0 h-full w-full max-w-xl bg-surface-base z-50 shadow-xl overflow-y-auto"
          >
            {/* Header */}
            <div className="sticky top-0 bg-surface-base border-b border-bdr p-4 flex items-center justify-between">
              <h2 className="text-lg font-semibold text-txt-primary">
                Visit Details
              </h2>
              <div className="flex items-center gap-2">
                {onEdit && (
                  <button
                    onClick={() => onEdit(visitId)}
                    className="p-2 hover:bg-hover rounded-lg"
                  >
                    <Edit className="w-4 h-4" />
                  </button>
                )}
                <button
                  onClick={handleShare}
                  className="p-2 hover:bg-hover rounded-lg"
                  title="Copy share link"
                >
                  <Share2 className="w-4 h-4" />
                </button>
                <button
                  onClick={onClose}
                  className="p-2 hover:bg-hover rounded-lg"
                >
                  <X className="w-5 h-5" />
                </button>
              </div>
            </div>

            {isLoading ? (
              <div className="p-6">Loading...</div>
            ) : visit ? (
              <div className="p-6 space-y-6">
                {/* Recruit Info */}
                <section>
                  <h3 className="text-2xl font-bold text-txt-primary mb-2">
                    {visit.recruitName}
                  </h3>
                  <div className="space-y-2 text-sm text-txt-secondary">
                    {visit.recruitEmail && (
                      <div className="flex items-center gap-2">
                        <Mail className="w-4 h-4" />
                        <a href={`mailto:${visit.recruitEmail}`} className="hover:text-interactive-primary">
                          {visit.recruitEmail}
                        </a>
                      </div>
                    )}
                    {visit.recruitPhone && (
                      <div className="flex items-center gap-2">
                        <Phone className="w-4 h-4" />
                        <span>{visit.recruitPhone}</span>
                      </div>
                    )}
                    {visit.recruitSchool && (
                      <div className="flex items-center gap-2">
                        <School className="w-4 h-4" />
                        <span>{visit.recruitSchool} {visit.recruitGradYear && `(${visit.recruitGradYear})`}</span>
                      </div>
                    )}
                  </div>
                </section>

                {/* Visit Time */}
                <section className="bg-violet-500/10 p-4 rounded-lg">
                  <div className="flex items-center gap-2 text-violet-500 mb-2">
                    <Calendar className="w-5 h-5" />
                    <span className="font-medium">
                      {new Date(visit.date).toLocaleDateString('en-US', {
                        weekday: 'long',
                        year: 'numeric',
                        month: 'long',
                        day: 'numeric',
                      })}
                    </span>
                  </div>
                  <div className="flex items-center gap-2 text-txt-secondary">
                    <Clock className="w-4 h-4" />
                    <span>{visit.startTime} - {visit.endTime}</span>
                  </div>
                </section>

                {/* Host Athlete */}
                {visit.hostAthlete && (
                  <section>
                    <h4 className="text-sm font-medium text-txt-secondary mb-2">Host Athlete</h4>
                    <div className="flex items-center gap-3 p-3 bg-surface-elevated rounded-lg">
                      <div className="w-10 h-10 rounded-full bg-violet-500/20 flex items-center justify-center">
                        <User className="w-5 h-5 text-violet-500" />
                      </div>
                      <div>
                        <p className="font-medium text-txt-primary">
                          {visit.hostAthlete.firstName} {visit.hostAthlete.lastName}
                        </p>
                        <p className="text-xs text-txt-secondary">Assigned host</p>
                      </div>
                    </div>
                  </section>
                )}

                {/* Schedule */}
                {(visit.scheduleContent || visit.schedulePdfUrl) && (
                  <section>
                    <h4 className="text-sm font-medium text-txt-secondary mb-2">Visit Schedule</h4>
                    {visit.scheduleType === 'richtext' && visit.scheduleContent ? (
                      <div className="prose prose-sm max-w-none p-4 bg-surface-elevated rounded-lg">
                        <RichTextDisplay content={visit.scheduleContent} />
                      </div>
                    ) : visit.schedulePdfUrl ? (
                      <a
                        href={visit.schedulePdfUrl}
                        target="_blank"
                        rel="noopener noreferrer"
                        className="flex items-center gap-2 p-4 bg-surface-elevated rounded-lg hover:bg-hover transition"
                      >
                        <FileText className="w-8 h-8 text-interactive-primary" />
                        <span className="text-interactive-primary">View Schedule PDF</span>
                      </a>
                    ) : null}
                  </section>
                )}

                {/* Notes */}
                {visit.notes && (
                  <section>
                    <h4 className="text-sm font-medium text-txt-secondary mb-2">Internal Notes</h4>
                    <p className="text-txt-primary whitespace-pre-wrap">{visit.notes}</p>
                  </section>
                )}

                {/* Delete action */}
                <div className="pt-4 border-t border-bdr">
                  {showDeleteConfirm ? (
                    <div className="flex items-center gap-2">
                      <span className="text-sm text-status-error">Delete this visit?</span>
                      <button
                        onClick={handleDelete}
                        className="px-3 py-1 bg-status-error text-white rounded text-sm"
                      >
                        Yes, delete
                      </button>
                      <button
                        onClick={() => setShowDeleteConfirm(false)}
                        className="px-3 py-1 text-txt-secondary text-sm"
                      >
                        Cancel
                      </button>
                    </div>
                  ) : (
                    <button
                      onClick={() => setShowDeleteConfirm(true)}
                      className="flex items-center gap-2 text-status-error hover:underline text-sm"
                    >
                      <Trash2 className="w-4 h-4" />
                      Delete visit
                    </button>
                  )}
                </div>
              </div>
            ) : (
              <div className="p-6 text-txt-secondary">Visit not found</div>
            )}
          </motion.div>
        </>
      )}
    </AnimatePresence>
  );
}
```

```typescript
import { useState } from 'react';
import { motion } from 'framer-motion';
import { Plus, Users, Calendar, Filter } from 'lucide-react';
import { useRecruitVisits } from '@v2/hooks/useRecruitVisits';
import { FeatureGuard } from '@v2/components/common';
import { RecruitVisitCard, RecruitVisitForm, VisitDetailPanel } from '@v2/components/recruiting';
import { Modal } from '@v2/components/ui/Modal';
import type { VisitStatus } from '@v2/types/recruiting';

export function RecruitingPage() {
  const [showForm, setShowForm] = useState(false);
  const [selectedVisitId, setSelectedVisitId] = useState<string | null>(null);
  const [editingVisitId, setEditingVisitId] = useState<string | null>(null);
  const [statusFilter, setStatusFilter] = useState<VisitStatus | 'all'>('all');

  const { data, isLoading } = useRecruitVisits(
    statusFilter !== 'all' ? { status: statusFilter } : undefined
  );

  const visits = data?.visits || [];

  const handleCreateSuccess = () => {
    setShowForm(false);
  };

  const handleEditSuccess = () => {
    setEditingVisitId(null);
    setSelectedVisitId(null);
  };

  return (
    <FeatureGuard featureId="recruiting">
      <div className="p-6 max-w-6xl mx-auto">
        {/* Header */}
        <div className="flex items-center justify-between mb-6">
          <div>
            <h1 className="text-2xl font-bold text-txt-primary flex items-center gap-2">
              <Users className="w-6 h-6 text-violet-500" />
              Recruiting
            </h1>
            <p className="text-txt-secondary mt-1">Manage recruit visits and schedules</p>
          </div>

          <button
            onClick={() => setShowForm(true)}
            className="flex items-center gap-2 px-4 py-2 bg-violet-500 text-white rounded-lg hover:bg-violet-600 transition"
          >
            <Plus className="w-4 h-4" />
            New Visit
          </button>
        </div>

        {/* Filters */}
        <div className="flex items-center gap-2 mb-6">
          <Filter className="w-4 h-4 text-txt-secondary" />
          {(['all', 'scheduled', 'completed', 'cancelled'] as const).map(status => (
            <button
              key={status}
              onClick={() => setStatusFilter(status)}
              className={`px-3 py-1.5 rounded-lg text-sm font-medium transition ${
                statusFilter === status
                  ? 'bg-violet-500 text-white'
                  : 'bg-surface text-txt-secondary hover:text-txt-primary'
              }`}
            >
              {status === 'all' ? 'All' : status.charAt(0).toUpperCase() + status.slice(1)}
            </button>
          ))}
        </div>

        {/* Visit List */}
        {isLoading ? (
          <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
            {[1, 2, 3].map(i => (
              <div key={i} className="h-40 bg-surface-elevated rounded-xl animate-pulse" />
            ))}
          </div>
        ) : visits.length === 0 ? (
          <div className="text-center py-12 bg-surface-elevated rounded-xl">
            <Users className="w-12 h-12 text-txt-tertiary mx-auto mb-4" />
            <h3 className="text-lg font-medium text-txt-primary mb-2">No recruit visits</h3>
            <p className="text-txt-secondary mb-4">
              {statusFilter !== 'all'
                ? `No ${statusFilter} visits found`
                : 'Create your first recruit visit to get started'}
            </p>
            {statusFilter === 'all' && (
              <button
                onClick={() => setShowForm(true)}
                className="inline-flex items-center gap-2 px-4 py-2 bg-violet-500 text-white rounded-lg"
              >
                <Plus className="w-4 h-4" />
                Create Visit
              </button>
            )}
          </div>
        ) : (
          <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
            {visits.map(visit => (
              <motion.div
                key={visit.id}
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
              >
                <RecruitVisitCard
                  visit={visit}
                  onClick={() => setSelectedVisitId(visit.id)}
                />
              </motion.div>
            ))}
          </div>
        )}

        {/* Create Form Modal */}
        <Modal isOpen={showForm} onClose={() => setShowForm(false)} title="New Recruit Visit">
          <RecruitVisitForm
            onSuccess={handleCreateSuccess}
            onCancel={() => setShowForm(false)}
          />
        </Modal>

        {/* Edit Form Modal */}
        <Modal
          isOpen={!!editingVisitId}
          onClose={() => setEditingVisitId(null)}
          title="Edit Recruit Visit"
        >
          {editingVisitId && (
            <RecruitVisitForm
              visit={visits.find(v => v.id === editingVisitId)}
              onSuccess={handleEditSuccess}
              onCancel={() => setEditingVisitId(null)}
            />
          )}
        </Modal>

        {/* Detail Panel */}
        <VisitDetailPanel
          visitId={selectedVisitId}
          onClose={() => setSelectedVisitId(null)}
          onEdit={(id) => {
            setSelectedVisitId(null);
            setEditingVisitId(id);
          }}
        />
      </div>
    </FeatureGuard>
  );
}

export default RecruitingPage;
```
