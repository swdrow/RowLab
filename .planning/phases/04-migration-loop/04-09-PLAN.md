```typescript
  const baseClasses = "p-0.5 rounded text-[10px] text-center border font-medium";

  return (
    <div
      className={`flex flex-col gap-0.5 p-1 ${isEditable ? 'cursor-pointer hover:bg-surface-hover rounded' : ''}`}
      onClick={isEditable ? onClick : undefined}
    >
      <div className={`${baseClasses} ${slotColors[morningSlot]}`} title="Morning">
        AM
      </div>
      <div className={`${baseClasses} ${slotColors[eveningSlot]}`} title="Evening">
        PM
      </div>
    </div>
  );
}
```

```typescript
import { useMemo } from 'react';
import type { AthleteAvailability } from '../../types/coach';
import { AvailabilityCell } from './AvailabilityCell';

interface AvailabilityGridProps {
  data: AthleteAvailability[];
  startDate: Date;
  numDays: number;
  onAthleteClick?: (athleteId: string) => void;
}

export function AvailabilityGrid({ data, startDate, numDays, onAthleteClick }: AvailabilityGridProps) {
  // Generate date range
  const dates = useMemo(() => {
    return Array.from({ length: numDays }, (_, i) => {
      const date = new Date(startDate);
      date.setDate(date.getDate() + i);
      return date;
    });
  }, [startDate, numDays]);

  const formatDateKey = (date: Date) => date.toISOString().split('T')[0];

  if (data.length === 0) {
    return (
      <div className="text-center py-12 text-txt-secondary">
        No athletes found. Athletes need to be added to the team first.
      </div>
    );
  }

  return (
    <div className="overflow-x-auto rounded-xl border border-bdr-primary">
      <div
        className="grid min-w-max"
        style={{
          gridTemplateColumns: `180px repeat(${numDays}, minmax(60px, 80px))`,
        }}
      >
        {/* Header row */}
        <div className="sticky left-0 z-10 bg-card-bg font-semibold p-3 border-b border-r border-bdr-primary text-txt-primary">
          Athlete
        </div>
        {dates.map((date) => (
          <div
            key={formatDateKey(date)}
            className="text-center p-2 border-b border-bdr-primary bg-card-bg"
          >
            <div className="text-[10px] text-txt-secondary uppercase">
              {date.toLocaleDateString('en-US', { weekday: 'short' })}
            </div>
            <div className="text-xs font-medium text-txt-primary">
              {date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
            </div>
          </div>
        ))}

        {/* Athlete rows */}
        {data.map((athlete) => (
          <>
            {/* Athlete name - sticky left */}
            <div
              key={`name-${athlete.athleteId}`}
              className={`sticky left-0 z-10 bg-card-bg p-3 border-b border-r border-bdr-primary font-medium text-txt-primary truncate ${
                onAthleteClick ? 'cursor-pointer hover:bg-surface-hover' : ''
              }`}
              onClick={() => onAthleteClick?.(athlete.athleteId)}
            >
              {athlete.athleteName}
            </div>

            {/* Availability cells */}
            {dates.map((date) => {
              const dateKey = formatDateKey(date);
              const dayData = athlete.dates.find((d) => d.date === dateKey);

              return (
                <div
                  key={`${athlete.athleteId}-${dateKey}`}
                  className="border-b border-bdr-primary/50"
                >
                  <AvailabilityCell
                    morningSlot={dayData?.morningSlot || 'NOT_SET'}
                    eveningSlot={dayData?.eveningSlot || 'NOT_SET'}
                  />
                </div>
              );
            })}
          </>
        ))}
      </div>
    </div>
  );
}
```

Key implementation:
- CSS Grid with dynamic columns based on date range
- Sticky first column (athlete names) using position: sticky
- Map athlete dates to grid cells, show NOT_SET if no data
- Optional onAthleteClick for navigating to edit athlete

```typescript
import { useState, useMemo } from 'react';
import type { AvailabilityDay, AvailabilitySlot } from '../../types/coach';

interface AvailabilityEditorProps {
  athleteName: string;
  initialData: AvailabilityDay[];
  startDate: Date;
  numDays: number;
  onSave: (availability: AvailabilityDay[]) => void;
  onCancel: () => void;
  isSaving: boolean;
}

const SLOT_OPTIONS: { value: AvailabilitySlot; label: string; color: string }[] = [
  { value: 'AVAILABLE', label: 'Available', color: 'bg-green-500' },
  { value: 'UNAVAILABLE', label: 'Unavailable', color: 'bg-red-500' },
  { value: 'MAYBE', label: 'Maybe', color: 'bg-yellow-500' },
  { value: 'NOT_SET', label: 'Not Set', color: 'bg-gray-500' },
];

export function AvailabilityEditor({
  athleteName,
  initialData,
  startDate,
  numDays,
  onSave,
  onCancel,
  isSaving,
}: AvailabilityEditorProps) {
  // Initialize state from initial data
  const [availability, setAvailability] = useState<Map<string, AvailabilityDay>>(() => {
    const map = new Map();
    initialData.forEach((day) => map.set(day.date, day));
    return map;
  });

  const dates = useMemo(() => {
    return Array.from({ length: numDays }, (_, i) => {
      const date = new Date(startDate);
      date.setDate(date.getDate() + i);
      return date;
    });
  }, [startDate, numDays]);

  const formatDateKey = (date: Date) => date.toISOString().split('T')[0];

  const updateSlot = (dateKey: string, slot: 'morning' | 'evening', value: AvailabilitySlot) => {
    setAvailability((prev) => {
      const next = new Map(prev);
      const existing = next.get(dateKey) || { date: dateKey, morningSlot: 'NOT_SET', eveningSlot: 'NOT_SET' };
      next.set(dateKey, {
        ...existing,
        [slot === 'morning' ? 'morningSlot' : 'eveningSlot']: value,
      });
      return next;
    });
  };

  const handleSave = () => {
    const data = Array.from(availability.values());
    onSave(data);
  };

  return (
    <div className="space-y-4">
      <div className="text-sm text-txt-secondary">
        Editing availability for <span className="font-medium text-txt-primary">{athleteName}</span>
      </div>

      <div className="overflow-x-auto">
        <div className="grid gap-2 min-w-max" style={{ gridTemplateColumns: `repeat(${numDays}, 100px)` }}>
          {dates.map((date) => {
            const dateKey = formatDateKey(date);
            const dayData = availability.get(dateKey) || { morningSlot: 'NOT_SET', eveningSlot: 'NOT_SET' };

            return (
              <div key={dateKey} className="p-2 rounded-lg border border-bdr-primary bg-surface">
                <div className="text-xs text-txt-secondary mb-2 text-center">
                  {date.toLocaleDateString('en-US', { weekday: 'short', month: 'short', day: 'numeric' })}
                </div>

                {/* Morning slot */}
                <div className="mb-2">
                  <label className="text-[10px] text-txt-secondary">Morning</label>
                  <select
                    value={dayData.morningSlot}
                    onChange={(e) => updateSlot(dateKey, 'morning', e.target.value as AvailabilitySlot)}
                    className="w-full px-2 py-1 text-xs rounded border border-bdr-primary bg-card-bg text-txt-primary"
                    disabled={isSaving}
                  >
                    {SLOT_OPTIONS.map((opt) => (
                      <option key={opt.value} value={opt.value}>{opt.label}</option>
                    ))}
                  </select>
                </div>

                {/* Evening slot */}
                <div>
                  <label className="text-[10px] text-txt-secondary">Evening</label>
                  <select
                    value={dayData.eveningSlot}
                    onChange={(e) => updateSlot(dateKey, 'evening', e.target.value as AvailabilitySlot)}
                    className="w-full px-2 py-1 text-xs rounded border border-bdr-primary bg-card-bg text-txt-primary"
                    disabled={isSaving}
                  >
                    {SLOT_OPTIONS.map((opt) => (
                      <option key={opt.value} value={opt.value}>{opt.label}</option>
                    ))}
                  </select>
                </div>
              </div>
            );
          })}
        </div>
      </div>

      {/* Actions */}
      <div className="flex justify-end gap-3 pt-4">
        <button
          onClick={onCancel}
          className="px-4 py-2 text-sm border border-bdr-primary rounded-lg hover:bg-surface-hover transition-colors"
          disabled={isSaving}
        >
          Cancel
        </button>
        <button
          onClick={handleSave}
          className="px-4 py-2 text-sm bg-accent-primary text-white rounded-lg hover:bg-accent-primary/90 transition-colors disabled:opacity-50"
          disabled={isSaving}
        >
          {isSaving ? 'Saving...' : 'Save Availability'}
        </button>
      </div>
    </div>
  );
}
```
