  onAddAthlete,

  onAddWorkout?: () => void;
  onApplyTemplate?: () => void;
  selectedDate?: Date;
}

export const CalendarEmptyState: React.FC<CalendarEmptyStateProps> = ({
  onAddWorkout,
  onApplyTemplate,
  selectedDate,
}) => {
  const dateStr = selectedDate?.toLocaleDateString('en-US', { month: 'long', day: 'numeric' });

  return (
    <EmptyState
      icon={Calendar}

      description={
        selectedDate

        onAddWorkout

        onApplyTemplate

              label: 'Apply Template',

  );
