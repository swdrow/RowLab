import { forwardRef, type ButtonHTMLAttributes, type ReactNode } from 'react';
import { motion, type HTMLMotionProps } from 'framer-motion';
import { Loader2 } from 'lucide-react';
import { SPRING_CONFIG, SPRING_FAST, usePrefersReducedMotion } from '@v2/utils/animations';
import { cn } from '@v2/utils/cn';

type ButtonVariant = 'primary' | 'secondary' | 'ghost' | 'danger' | 'outline';
type ButtonSize = 'sm' | 'md' | 'lg';

interface ButtonProps extends Omit<HTMLMotionProps<'button'>, 'ref'> {
  variant?: ButtonVariant;
  size?: ButtonSize;
  loading?: boolean;
  leftIcon?: ReactNode;
  rightIcon?: ReactNode;
  fullWidth?: boolean;
}

const variantStyles: Record<ButtonVariant, string> = {
  primary: `
    bg-interactive-primary text-white border-interactive-primary
    hover:bg-interactive-hover hover:shadow-[0_0_20px_rgba(0,112,243,0.4)]
    active:scale-[0.98]
    disabled:bg-interactive-disabled disabled:border-interactive-disabled
    disabled:text-txt-tertiary disabled:shadow-none disabled:cursor-not-allowed
  `,
  secondary: `
    bg-surface-elevated text-txt-primary border-bdr-default
    hover:bg-hover hover:border-bdr-brand
    active:scale-[0.98]
    disabled:bg-surface disabled:text-txt-tertiary disabled:cursor-not-allowed
  `,
  ghost: `
    bg-transparent text-txt-secondary border-transparent
    hover:bg-hover hover:text-txt-primary
    active:bg-surface-elevated
    disabled:text-txt-tertiary disabled:cursor-not-allowed
  `,
  danger: `
    bg-status-error/10 text-status-error border-status-error/20
    hover:bg-status-error/20 hover:border-status-error/40
    active:scale-[0.98]
    disabled:bg-surface disabled:text-txt-tertiary disabled:cursor-not-allowed
  `,
  outline: `
    bg-transparent text-txt-primary border-bdr-default
    hover:bg-hover hover:border-bdr-brand
    active:scale-[0.98]
    disabled:text-txt-tertiary disabled:cursor-not-allowed
  `,
};

const sizeStyles: Record<ButtonSize, string> = {
  sm: 'px-3 py-1.5 text-sm gap-1.5 rounded-lg',
  md: 'px-4 py-2.5 text-sm gap-2 rounded-xl',
  lg: 'px-6 py-3 text-base gap-2 rounded-xl',
};

export const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      variant = 'primary',
      size = 'md',
      loading = false,
      leftIcon,
      rightIcon,
      fullWidth = false,
      disabled,
      className,
      children,
      ...props
    },
    ref
  ) => {
    const prefersReducedMotion = usePrefersReducedMotion();

    return (
      <motion.button
        ref={ref}
        whileHover={prefersReducedMotion ? {} : { scale: 1.02 }}
        whileTap={prefersReducedMotion ? {} : { scale: 0.98 }}
        transition={prefersReducedMotion ? { duration: 0 } : SPRING_FAST}
        disabled={disabled || loading}
        className={cn(
          'inline-flex items-center justify-center font-medium transition-colors',
          'border focus:outline-none focus-visible:ring-2 focus-visible:ring-interactive-primary focus-visible:ring-offset-2',
          variantStyles[variant],
          sizeStyles[size],
          fullWidth && 'w-full',
          className
        )}
        {...props}
      >
        {loading ? (
          <Loader2 className="w-4 h-4 animate-spin" />
        ) : leftIcon ? (
          leftIcon
        ) : null}
        {children}
        {rightIcon && !loading && rightIcon}
      </motion.button>
    );
  }
);

Button.displayName = 'Button';

// IconButton for icon-only buttons
export const IconButton = forwardRef<HTMLButtonElement, Omit<ButtonProps, 'children'>>(
  ({ variant = 'ghost', size = 'md', className, ...props }, ref) => {
    const iconSizes = { sm: 'w-8 h-8', md: 'w-10 h-10', lg: 'w-12 h-12' };
    return (
      <Button
        ref={ref}
        variant={variant}
        className={cn(iconSizes[size], 'p-0', className)}
        {...props}
      />
    );
  }
);

IconButton.displayName = 'IconButton';
```

- Icon slots (left, right)
- Focus-visible ring for keyboard navigation
- Reduced motion support

import { motion } from 'framer-motion';
import { SPRING_CONFIG, usePrefersReducedMotion } from '@v2/utils/animations';
import { cn } from '@v2/utils/cn';

interface ToggleProps {
  enabled: boolean;
  onChange: (enabled: boolean) => void;
  disabled?: boolean;
  size?: 'sm' | 'md';
  label?: string;
  description?: string;
}

export const Toggle: React.FC<ToggleProps> = ({
  enabled,
  onChange,
  disabled = false,
  size = 'md',
  label,
  description,
}) => {
  const prefersReducedMotion = usePrefersReducedMotion();

  const sizes = {
    sm: {
      track: 'w-10 h-6',
      thumb: 'w-4 h-4',
      translate: enabled ? 'translateX(18px)' : 'translateX(2px)',
    },
    md: {
      track: 'w-12 h-7',
      thumb: 'w-5 h-5',
      translate: enabled ? 'translateX(22px)' : 'translateX(2px)',
    },
  };

  const s = sizes[size];

  const handleClick = () => {
    if (!disabled) {
      onChange(!enabled);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      handleClick();
    }
  };

  return (
    <div className="flex items-center gap-3">
      <button
        type="button"
        role="switch"
        aria-checked={enabled}
        aria-label={label}
        onClick={handleClick}
        onKeyDown={handleKeyDown}
        disabled={disabled}
        className={cn(
          'relative rounded-full transition-colors',
          'focus:outline-none focus-visible:ring-2 focus-visible:ring-interactive-primary focus-visible:ring-offset-2',
          s.track,
          enabled
            ? 'bg-interactive-primary shadow-[0_0_12px_rgba(0,112,243,0.4)]'
            : 'bg-surface-elevated border border-bdr-subtle',
          disabled && 'opacity-50 cursor-not-allowed'
        )}
      >
        <motion.span
          animate={{
            x: enabled ? (size === 'sm' ? 18 : 22) : 2,
          }}
          transition={prefersReducedMotion ? { duration: 0 } : SPRING_CONFIG}
          className={cn(
            'absolute top-1 rounded-full shadow',
            s.thumb,
            enabled ? 'bg-white' : 'bg-txt-tertiary'
          )}
        />
      </button>
      {(label || description) && (
        <div className="flex flex-col">
          {label && <span className="text-sm font-medium text-txt-primary">{label}</span>}
          {description && <span className="text-xs text-txt-secondary">{description}</span>}
        </div>
      )}
    </div>
  );
};
```

Features:
- Spring animation for thumb movement
- Glow effect when enabled
- Two sizes (sm, md)
- Keyboard accessible (Enter, Space)
- ARIA attributes (role="switch", aria-checked)
- Optional label and description
- Disabled state
- Focus-visible ring

```typescript
import { Fragment, type ReactNode } from 'react';
import { Dialog, Transition } from '@headlessui/react';
import { motion, AnimatePresence } from 'framer-motion';
import { X } from 'lucide-react';
import { SPRING_CONFIG, TRANSITION_DURATION, usePrefersReducedMotion } from '@v2/utils/animations';
import { cn } from '@v2/utils/cn';
import { IconButton } from './Button';

interface ModalProps {
  open: boolean;
  onClose: () => void;
  title?: string;
  description?: string;
  children: ReactNode;
  size?: 'sm' | 'md' | 'lg' | 'xl';
  showCloseButton?: boolean;
}

const sizeStyles = {
  sm: 'max-w-sm',
  md: 'max-w-md',
  lg: 'max-w-lg',
  xl: 'max-w-xl',
};

export const Modal: React.FC<ModalProps> = ({
  open,
  onClose,
  title,
  description,
  children,
  size = 'md',
  showCloseButton = true,
}) => {
  const prefersReducedMotion = usePrefersReducedMotion();

  return (
    <AnimatePresence>
      {open && (
        <Dialog static open={open} onClose={onClose} className="relative z-50">
          {/* Backdrop */}
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            transition={{ duration: prefersReducedMotion ? 0 : TRANSITION_DURATION.normal }}
            className="fixed inset-0 bg-black/50 backdrop-blur-sm"
            aria-hidden="true"
          />

          {/* Full-screen container for centering */}
          <div className="fixed inset-0 flex items-center justify-center p-4">
            <Dialog.Panel
              as={motion.div}
              initial={{ opacity: 0, y: 20, scale: 0.95 }}
              animate={{ opacity: 1, y: 0, scale: 1 }}
              exit={{ opacity: 0, y: 20, scale: 0.95 }}
              transition={prefersReducedMotion ? { duration: 0 } : SPRING_CONFIG}
              className={cn(
                'w-full rounded-2xl bg-surface border border-bdr-subtle shadow-xl',
                'max-h-[85vh] overflow-hidden flex flex-col',
                sizeStyles[size]
              )}
            >
              {/* Header */}
              {(title || showCloseButton) && (
                <div className="flex items-center justify-between px-6 py-4 border-b border-bdr-subtle">
                  <div>
                    {title && (
                      <Dialog.Title className="text-lg font-semibold text-txt-primary">
                        {title}
                      </Dialog.Title>
                    )}
                    {description && (
                      <Dialog.Description className="mt-1 text-sm text-txt-secondary">
                        {description}
                      </Dialog.Description>
                    )}
                  </div>
                  {showCloseButton && (
                    <IconButton
                      variant="ghost"
                      size="sm"
                      onClick={onClose}
                      aria-label="Close modal"
                    >
                      <X className="w-5 h-5" />
                    </IconButton>
                  )}
                </div>
              )}

              {/* Content */}
              <div className="flex-1 overflow-y-auto px-6 py-4">
                {children}
              </div>
            </Dialog.Panel>
          </div>
        </Dialog>
      )}
    </AnimatePresence>
  );
};

// Modal footer for action buttons
export const ModalFooter: React.FC<{ children: ReactNode; className?: string }> = ({
  children,
  className,
}) => (
  <div className={cn('flex items-center justify-end gap-3 px-6 py-4 border-t border-bdr-subtle', className)}>
    {children}
  </div>
);
```

- Headless UI Dialog for focus trapping and accessibility
- Framer Motion for animations
- Backdrop blur with fade
- Panel slide+fade+scale animation
- Four sizes
- Optional title, description, close button
- Scrollable content area
- ModalFooter component for actions
- Reduced motion support
- Escape key closes modal (built into Headless UI)
