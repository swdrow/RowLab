```typescript
// Live erg data from C2 Logbook polling
export interface LiveErgData {
  athleteId: string;
  athleteName: string;
  distance: number; // meters
  time: number; // seconds
  pace: number; // seconds per 500m
  watts: number;
  strokeRate: number; // strokes per minute
  heartRate?: number;
  strokeCount?: number;
  status: 'active' | 'resting' | 'finished' | 'pending';
  lastUpdated: string; // ISO timestamp
}

// Aggregated session data
export interface LiveSessionData {
  sessionId: string;
  sessionName: string;
  activePieceId?: string;
  activePieceName?: string;
  athletes: LiveErgData[];
  startedAt: string;
  sessionCode: string;
}

// Athlete status for display
export type AthleteErgStatus = 'active' | 'resting' | 'finished' | 'pending';

// View mode toggle
export type LiveViewMode = 'leaderboard' | 'grid';

// Polling configuration
export interface PollingConfig {
  interval: number; // milliseconds
  enabled: boolean;
}

// Default polling interval (5 seconds per RESEARCH.md recommendation)
export const DEFAULT_POLLING_INTERVAL = 5000;

// Format helpers
export function formatPace(seconds: number): string {
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  const tenths = Math.floor((seconds % 1) * 10);
  return `${mins}:${secs.toString().padStart(2, '0')}.${tenths}`;
}

export function formatTime(seconds: number): string {
  const hours = Math.floor(seconds / 3600);
  const mins = Math.floor((seconds % 3600) / 60);
  const secs = Math.floor(seconds % 60);

  if (hours > 0) {
    return `${hours}:${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
  }
  return `${mins}:${secs.toString().padStart(2, '0')}`;
}

export function formatDistance(meters: number): string {
  if (meters >= 1000) {
    return `${(meters / 1000).toFixed(1)}km`;
  }
  return `${meters}m`;
}

// Status colors
export const STATUS_COLORS: Record<AthleteErgStatus, string> = {
  active: 'text-green-500',
  resting: 'text-amber-500',
  finished: 'text-blue-500',
  pending: 'text-txt-muted',
};
```

```typescript
import { useQuery } from '@tanstack/react-query';
import { useAuthStore } from '../../../../stores/authStore';
import type { LiveSessionData, PollingConfig } from '../../../types/live-erg';
import { DEFAULT_POLLING_INTERVAL } from '../../../types/live-erg';

// Query key for live erg data
export const liveErgKeys = {
  session: (sessionId: string) => ['live-erg', sessionId] as const,
};

interface UseLiveErgPollingOptions {
  sessionId: string;
  config?: Partial<PollingConfig>;
}

export function useLiveErgPolling({
  sessionId,
  config = {},
}: UseLiveErgPollingOptions) {
  const { accessToken, teamId } = useAuthStore();

  const pollingConfig: PollingConfig = {
    interval: config.interval ?? DEFAULT_POLLING_INTERVAL,
    enabled: config.enabled ?? true,
  };

  return useQuery({
    queryKey: liveErgKeys.session(sessionId),
    queryFn: async (): Promise<LiveSessionData> => {
      const response = await fetch(`/api/v1/sessions/${sessionId}/live-data`, {
        headers: {
          Authorization: `Bearer ${accessToken}`,
          'X-Team-ID': teamId || '',
        },
      });

      if (!response.ok) {
        throw new Error('Failed to fetch live erg data');
      }

      return response.json();
    },
    enabled: !!accessToken && !!teamId && !!sessionId && pollingConfig.enabled,
    // Poll at configured interval when enabled
    refetchInterval: pollingConfig.enabled ? pollingConfig.interval : false,
    // Continue polling in background when tab is inactive
    refetchIntervalInBackground: true,
    // Data considered fresh for half the polling interval
    staleTime: pollingConfig.interval / 2,
    // Don't refetch on window focus (polling handles freshness)
    refetchOnWindowFocus: false,
  });
}

// Hook to control polling state
export function usePollingControls(initialEnabled = true, initialInterval = DEFAULT_POLLING_INTERVAL) {
  const [enabled, setEnabled] = useState(initialEnabled);
  const [interval, setInterval] = useState(initialInterval);

  const pause = () => setEnabled(false);
  const resume = () => setEnabled(true);
  const setPollingInterval = (ms: number) => setInterval(Math.max(1000, ms)); // Min 1 second

  return {
    enabled,
    interval,
    pause,
    resume,
    setPollingInterval,
    config: { enabled, interval },
  };
}
```

```typescript
import { motion } from 'framer-motion';
import type { LiveErgData } from '../../../types/live-erg';
import { formatPace, formatDistance, STATUS_COLORS } from '../../../types/live-erg';

interface RankedLeaderboardProps {
  athletes: LiveErgData[];
  sortBy?: 'pace' | 'distance' | 'watts';
}

export function RankedLeaderboard({ athletes, sortBy = 'pace' }: RankedLeaderboardProps) {
  // Sort athletes (active first, then by metric)
  const sorted = [...athletes].sort((a, b) => {
    // Pending/resting at bottom
    if (a.status === 'pending' && b.status !== 'pending') return 1;
    if (a.status !== 'pending' && b.status === 'pending') return -1;

    // Sort by selected metric
    switch (sortBy) {
      case 'distance':
        return b.distance - a.distance;
      case 'watts':
        return b.watts - a.watts;
      case 'pace':
      default:
        return a.pace - b.pace; // Lower pace is better
    }
  });

  return (
    <div className="bg-surface-elevated rounded-lg border border-bdr-default overflow-hidden">
      <table className="w-full">
        <thead>
          <tr className="border-b border-bdr-default bg-surface-default">
            <th className="px-4 py-3 text-left text-xs font-medium text-txt-muted">Rank</th>
            <th className="px-4 py-3 text-left text-xs font-medium text-txt-muted">Athlete</th>
            <th className="px-4 py-3 text-right text-xs font-medium text-txt-muted">Pace</th>
            <th className="px-4 py-3 text-right text-xs font-medium text-txt-muted">Watts</th>
            <th className="px-4 py-3 text-right text-xs font-medium text-txt-muted">Distance</th>
            <th className="px-4 py-3 text-right text-xs font-medium text-txt-muted">Rate</th>
            <th className="px-4 py-3 text-center text-xs font-medium text-txt-muted">Status</th>
          </tr>
        </thead>
        <tbody>
          {sorted.map((athlete, index) => (
            <motion.tr
              key={athlete.athleteId}
              layout
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              className="border-b border-bdr-default last:border-0 hover:bg-surface-hover"
            >
              <td className="px-4 py-3">
                <span className={`w-6 h-6 inline-flex items-center justify-center rounded-full
                  ${index < 3 ? 'bg-accent-primary/20 text-accent-primary font-bold' : 'text-txt-secondary'}`}>
                  {athlete.status !== 'pending' ? index + 1 : '-'}
                </span>
              </td>
              <td className="px-4 py-3 font-medium text-txt-primary">{athlete.athleteName}</td>
              <td className="px-4 py-3 text-right font-mono text-txt-primary">
                {athlete.status !== 'pending' ? formatPace(athlete.pace) : '-'}
              </td>
              <td className="px-4 py-3 text-right font-mono text-txt-primary">
                {athlete.status !== 'pending' ? athlete.watts : '-'}
              </td>
              <td className="px-4 py-3 text-right font-mono text-txt-primary">
                {athlete.status !== 'pending' ? formatDistance(athlete.distance) : '-'}
              </td>
              <td className="px-4 py-3 text-right font-mono text-txt-primary">
                {athlete.status !== 'pending' ? `${athlete.strokeRate}` : '-'}
              </td>
              <td className="px-4 py-3 text-center">
                <span className={`text-sm capitalize ${STATUS_COLORS[athlete.status]}`}>
                  {athlete.status}
                </span>
              </td>
            </motion.tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
```

```typescript
import { useState } from 'react';
import { Play, Pause, SquaresFour, ListBullets, Gear, Users } from '@phosphor-icons/react';
import { useLiveErgPolling, usePollingControls } from '../hooks/useLiveErgPolling';
import { AthleteErgCard } from './AthleteErgCard';
import { RankedLeaderboard } from './RankedLeaderboard';
import type { LiveViewMode } from '../../../types/live-erg';

interface LiveErgDashboardProps {
  sessionId: string;
  sessionName?: string;
  sessionCode?: string;
  targetPace?: number;
}

export function LiveErgDashboard({
  sessionId,
  sessionName,
  sessionCode,
  targetPace,
}: LiveErgDashboardProps) {
  const [viewMode, setViewMode] = useState<LiveViewMode>('leaderboard');
  const [showTargets, setShowTargets] = useState(true);
  const polling = usePollingControls(true, 5000);

  const { data, isLoading, isError, dataUpdatedAt } = useLiveErgPolling({
    sessionId,
    config: polling.config,
  });

  // Separate active athletes from pending
  const activeAthletes = data?.athletes.filter((a) => a.status !== 'pending') || [];
  const pendingAthletes = data?.athletes.filter((a) => a.status === 'pending') || [];

  if (isLoading && !data) {
    return (
      <div className="flex items-center justify-center h-64">
        <div className="animate-spin w-8 h-8 border-2 border-accent-primary border-t-transparent rounded-full" />
      </div>
    );
  }

  if (isError) {
    return (
      <div className="bg-red-500/10 border border-red-500/20 rounded-lg p-4 text-red-500">
        Failed to load live erg data. Please try again.
      </div>
    );
  }

  return (
    <div className="space-y-4">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h2 className="text-xl font-semibold text-txt-primary">
            {sessionName || 'Live Session'}
          </h2>
          {sessionCode && (
            <p className="text-sm text-txt-secondary">
              Session Code: <span className="font-mono font-medium">{sessionCode}</span>
            </p>
          )}
        </div>

        {/* Controls */}
        <div className="flex items-center gap-2">
          {/* Polling control */}
          <button
            onClick={polling.enabled ? polling.pause : polling.resume}
            className={`p-2 rounded-lg border ${
              polling.enabled
                ? 'bg-green-500/10 border-green-500/20 text-green-500'
                : 'bg-surface-default border-bdr-default text-txt-secondary'
            }`}
            title={polling.enabled ? 'Pause updates' : 'Resume updates'}
          >
            {polling.enabled ? <Pause className="w-5 h-5" /> : <Play className="w-5 h-5" />}
          </button>

          {/* View toggle */}
          <div className="flex items-center border border-bdr-default rounded-lg overflow-hidden">
            <button
              onClick={() => setViewMode('leaderboard')}
              className={`p-2 ${viewMode === 'leaderboard' ? 'bg-accent-primary/10 text-accent-primary' : 'text-txt-secondary'}`}
              title="Leaderboard view"
            >
              <ListBullets className="w-5 h-5" />
            </button>
            <button
              onClick={() => setViewMode('grid')}
              className={`p-2 ${viewMode === 'grid' ? 'bg-accent-primary/10 text-accent-primary' : 'text-txt-secondary'}`}
              title="Grid view"
            >
              <SquaresFour className="w-5 h-5" />
            </button>
          </div>

          {/* Target toggle */}
          <button
            onClick={() => setShowTargets(!showTargets)}
            className={`p-2 rounded-lg border ${
              showTargets
                ? 'bg-accent-primary/10 border-accent-primary/20 text-accent-primary'
                : 'bg-surface-default border-bdr-default text-txt-secondary'
            }`}
            title="Toggle target display"
          >
            <Gear className="w-5 h-5" />
          </button>
        </div>
      </div>

      {/* Status bar */}
      <div className="flex items-center justify-between text-sm text-txt-secondary">
        <span className="flex items-center gap-2">
          <Users className="w-4 h-4" />
          {activeAthletes.length} active, {pendingAthletes.length} waiting
        </span>
        <span>
          Last updated: {dataUpdatedAt ? new Date(dataUpdatedAt).toLocaleTimeString() : 'Never'}
        </span>
      </div>

      {/* Active athletes */}
      {viewMode === 'leaderboard' ? (
        <RankedLeaderboard athletes={activeAthletes} />
      ) : (
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
          {activeAthletes.map((athlete, index) => (
            <AthleteErgCard
              key={athlete.athleteId}
              data={athlete}
              rank={index + 1}
              showTargets={showTargets}
              targetPace={targetPace}
            />
          ))}
        </div>
      )}

      {/* Pending athletes */}
      {pendingAthletes.length > 0 && (
        <div>
          <h3 className="text-sm font-medium text-txt-secondary mb-2">Waiting to Join</h3>
          <div className="flex flex-wrap gap-2">
            {pendingAthletes.map((athlete) => (
              <span
                key={athlete.athleteId}
                className="px-3 py-1 rounded-full bg-surface-default border border-bdr-default text-txt-muted text-sm"
              >
                {athlete.athleteName}
              </span>
            ))}
          </div>
        </div>
      )}
    </div>
  );
}
```
