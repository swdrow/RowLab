```javascript
import express from 'express';
import { PrismaClient } from '@prisma/client';
import { body, query, param, validationResult } from 'express-validator';
import { authenticateToken, requireTeam } from '../middleware/auth.js';

const router = express.Router();
const prisma = new PrismaClient();

// All routes require authentication and team context
router.use(authenticateToken);
router.use(requireTeam);

// Generate unique session code (6 alphanumeric characters)
function generateSessionCode() {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789';
  let code = '';
  for (let i = 0; i < 6; i++) {
    code += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return code;
}

// GET /api/v1/sessions - List sessions with filters
router.get(
  '/',
  [
    query('type').optional().isIn(['ERG', 'ROW', 'LIFT', 'RUN', 'CROSS_TRAIN', 'RECOVERY']),
    query('status').optional().isIn(['PLANNED', 'ACTIVE', 'COMPLETED', 'CANCELLED']),
    query('startDate').optional().isISO8601(),
    query('endDate').optional().isISO8601(),
  ],
  async (req, res) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }

    const { type, status, startDate, endDate } = req.query;
    const teamId = req.teamId;

    const where = { teamId };
    if (type) where.type = type;
    if (status) where.status = status;
    if (startDate || endDate) {
      where.date = {};
      if (startDate) where.date.gte = new Date(startDate);
      if (endDate) where.date.lte = new Date(endDate);
    }

    try {
      const [sessions, total] = await Promise.all([
        prisma.session.findMany({
          where,
          include: { pieces: { orderBy: { order: 'asc' } } },
          orderBy: { date: 'desc' },
        }),
        prisma.session.count({ where }),
      ]);

      res.json({ sessions, total });
    } catch (error) {
      console.error('Error fetching sessions:', error);
      res.status(500).json({ error: 'Failed to fetch sessions' });
    }
  }
);

// GET /api/v1/sessions/active - Get currently active session for team
router.get('/active', async (req, res) => {
  const teamId = req.teamId;

  try {
    const session = await prisma.session.findFirst({
      where: {
        teamId,
        status: 'ACTIVE',
      },
      include: { pieces: { orderBy: { order: 'asc' } } },
    });

    if (!session) {
      return res.status(404).json({ error: 'No active session' });
    }

    res.json(session);
  } catch (error) {
    console.error('Error fetching active session:', error);
    res.status(500).json({ error: 'Failed to fetch active session' });
  }
});

// GET /api/v1/sessions/:id - Get single session
router.get('/:id', param('id').isString(), async (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }

  const { id } = req.params;
  const teamId = req.teamId;

  try {
    const session = await prisma.session.findFirst({
      where: { id, teamId },
      include: { pieces: { orderBy: { order: 'asc' } } },
    });

    if (!session) {
      return res.status(404).json({ error: 'Session not found' });
    }

    res.json(session);
  } catch (error) {
    console.error('Error fetching session:', error);
    res.status(500).json({ error: 'Failed to fetch session' });
  }
});

// POST /api/v1/sessions - Create session with pieces
router.post(
  '/',
  [
    body('name').isString().trim().notEmpty(),
    body('type').isIn(['ERG', 'ROW', 'LIFT', 'RUN', 'CROSS_TRAIN', 'RECOVERY']),
    body('date').isISO8601(),
    body('startTime').optional().isISO8601(),
    body('endTime').optional().isISO8601(),
    body('recurrenceRule').optional().isString(),
    body('notes').optional().isString(),
    body('pieces').optional().isArray(),
    body('athleteVisibility').optional().isBoolean(),
  ],
  async (req, res) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }

    const teamId = req.teamId;
    const userId = req.user.id;
    const { name, type, date, startTime, endTime, recurrenceRule, notes, pieces, athleteVisibility } = req.body;

    try {
      const session = await prisma.session.create({
        data: {
          name,
          type,
          date: new Date(date),
          startTime: startTime ? new Date(startTime) : null,
          endTime: endTime ? new Date(endTime) : null,
          recurrenceRule,
          notes,
          status: 'PLANNED',
          teamId,
          createdById: userId,
          athleteVisibility: athleteVisibility ?? true,
          pieces: pieces
            ? {
                create: pieces.map((piece, index) => ({
                  segment: piece.segment || 'MAIN',
                  name: piece.name,
                  description: piece.description,
                  order: piece.order ?? index,
                  distance: piece.distance,
                  duration: piece.duration,
                  targetSplit: piece.targetSplit,
                  targetRate: piece.targetRate,
                  targetWatts: piece.targetWatts,
                  targetHRZone: piece.targetHRZone,
                  targetRPE: piece.targetRPE,
                  notes: piece.notes,
                  boatClass: piece.boatClass,
                  sets: piece.sets,
                  reps: piece.reps,
                })),
              }
            : undefined,
        },
        include: { pieces: { orderBy: { order: 'asc' } } },
      });

      res.status(201).json(session);
    } catch (error) {
      console.error('Error creating session:', error);
      res.status(500).json({ error: 'Failed to create session' });
    }
  }
);

// PATCH /api/v1/sessions/:id - Update session
router.patch(
  '/:id',
  [
    param('id').isString(),
    body('name').optional().isString().trim().notEmpty(),
    body('type').optional().isIn(['ERG', 'ROW', 'LIFT', 'RUN', 'CROSS_TRAIN', 'RECOVERY']),
    body('date').optional().isISO8601(),
    body('status').optional().isIn(['PLANNED', 'ACTIVE', 'COMPLETED', 'CANCELLED']),
    body('athleteVisibility').optional().isBoolean(),
  ],
  async (req, res) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }

    const { id } = req.params;
    const teamId = req.teamId;
    const updateData = { ...req.body };

    // Convert date strings to Date objects
    if (updateData.date) updateData.date = new Date(updateData.date);
    if (updateData.startTime) updateData.startTime = new Date(updateData.startTime);
    if (updateData.endTime) updateData.endTime = new Date(updateData.endTime);

    // Generate session code when starting a live session
    if (updateData.status === 'ACTIVE') {
      const existing = await prisma.session.findFirst({ where: { id, teamId } });
      if (existing && !existing.sessionCode) {
        // Generate unique code
        let code;
        let isUnique = false;
        while (!isUnique) {
          code = generateSessionCode();
          const existingCode = await prisma.session.findUnique({ where: { sessionCode: code } });
          if (!existingCode) isUnique = true;
        }
        updateData.sessionCode = code;
      }
    }

    try {
      const session = await prisma.session.update({
        where: { id, teamId },
        data: updateData,
        include: { pieces: { orderBy: { order: 'asc' } } },
      });

      res.json(session);
    } catch (error) {
      if (error.code === 'P2025') {
        return res.status(404).json({ error: 'Session not found' });
      }
      console.error('Error updating session:', error);
      res.status(500).json({ error: 'Failed to update session' });
    }
  }
);

// DELETE /api/v1/sessions/:id - Delete session and pieces
router.delete('/:id', param('id').isString(), async (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }

  const { id } = req.params;
  const teamId = req.teamId;

  try {
    // Delete pieces first (cascade), then session
    await prisma.piece.deleteMany({ where: { sessionId: id } });
    await prisma.session.delete({ where: { id, teamId } });

    res.status(204).send();
  } catch (error) {
    if (error.code === 'P2025') {
      return res.status(404).json({ error: 'Session not found' });
    }
    console.error('Error deleting session:', error);
    res.status(500).json({ error: 'Failed to delete session' });
  }
});

// POST /api/v1/sessions/join/:code - Join session by code (athlete endpoint)
router.post('/join/:code', param('code').isString().isLength({ min: 6, max: 6 }), async (req, res) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    return res.status(400).json({ errors: errors.array() });
  }

  const { code } = req.params;

  try {
    const session = await prisma.session.findUnique({
      where: { sessionCode: code.toUpperCase() },
      include: { pieces: { orderBy: { order: 'asc' } } },
    });

    if (!session) {
      return res.status(404).json({ error: 'Invalid session code' });
    }

    if (session.status !== 'ACTIVE') {
      return res.status(400).json({ error: 'Session is not currently active' });
    }

    res.json(session);
  } catch (error) {
    console.error('Error joining session:', error);
    res.status(500).json({ error: 'Failed to join session' });
  }
});

export default router;
```
