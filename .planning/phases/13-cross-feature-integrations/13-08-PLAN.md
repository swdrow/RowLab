```javascript
import express from 'express';
import { PrismaClient } from '@prisma/client';
import { query, validationResult } from 'express-validator';
import { authenticateToken, requireTeam } from '../middleware/auth.js';

const router = express.Router();
const prisma = new PrismaClient();

router.use(authenticateToken);
router.use(requireTeam);

// GET /api/v1/activities/unified - Get unified activity feed
router.get(
  '/unified',
  [
    query('athleteId').optional().isString(),
    query('cursor').optional().isString(),
    query('limit').optional().isInt({ min: 1, max: 50 }).toInt(),
  ],
  async (req, res) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({ errors: errors.array() });
    }

    const teamId = req.teamId;
    const { athleteId, cursor, limit = 20 } = req.query;

    try {
      const activities = [];
      const cursorDate = cursor ? new Date(cursor) : new Date();

      // Base where clause
      const athleteWhere = athleteId ? { athleteId } : { teamId };

      // Fetch erg tests
      const ergTests = await prisma.ergTest.findMany({
        where: {
          ...athleteWhere,
          testDate: { lt: cursorDate },
        },
        include: {
          athlete: { select: { id: true, firstName: true, lastName: true } },
        },
        orderBy: { testDate: 'desc' },
        take: limit,
      });

      ergTests.forEach((test) => {
        activities.push({
          id: `erg-${test.id}`,
          type: 'erg_test',
          title: `${test.testType} Test`,
          description: `Completed ${test.testType} erg test`,
          date: test.testDate.toISOString(),
          athleteId: test.athleteId,
          athleteName: `${test.athlete.firstName} ${test.athlete.lastName}`,
          metadata: {
            testType: test.testType,
            time: test.time,
            distance: test.distance,
            watts: test.watts,
          },
        });
      });

      // Fetch session attendance
      const attendance = await prisma.sessionAttendance.findMany({
        where: {
          session: { teamId },
          ...(athleteId && { athleteId }),
          createdAt: { lt: cursorDate },
        },
        include: {
          session: { select: { id: true, name: true, type: true, date: true } },
          athlete: { select: { id: true, firstName: true, lastName: true } },
        },
        orderBy: { createdAt: 'desc' },
        take: limit,
      });

      attendance.forEach((record) => {
        activities.push({
          id: `session-${record.id}`,
          type: 'session_participation',
          title: record.session.name,
          description: `Participated in ${record.session.type.toLowerCase()} session`,
          date: record.session.date.toISOString(),
          athleteId: record.athleteId,
          athleteName: `${record.athlete.firstName} ${record.athlete.lastName}`,
          metadata: {
            sessionId: record.sessionId,
            sessionType: record.session.type,
            sessionName: record.session.name,
            participationPercent: record.participationPercent,
          },
        });
      });

      // Fetch race results (if RaceEntry exists)
      try {
        const raceEntries = await prisma.raceEntry.findMany({
          where: {
            race: { regatta: { teamId } },
            ...(athleteId && {
              OR: [
                { lineup: { assignments: { some: { athleteId } } } },
              ],
            }),
            race: { date: { lt: cursorDate } },
          },
          include: {
            race: {
              include: {
                event: true,
                regatta: true,
              },
            },
          },
          orderBy: { race: { date: 'desc' } },
          take: limit,
        });

        raceEntries.forEach((entry) => {
          if (entry.finishTime) {
            activities.push({
              id: `race-${entry.id}`,
              type: 'race_result',
              title: `${entry.race.event.name} - ${entry.race.regatta.name}`,
              description: `Finished ${entry.finishPlace ? `#${entry.finishPlace}` : ''}`,
              date: entry.race.date.toISOString(),
              athleteId: athleteId || '',
              metadata: {
                regattaName: entry.race.regatta.name,
                eventName: entry.race.event.name,
                boatClass: entry.race.event.boatClass,
                place: entry.finishPlace || 0,
                time: entry.finishTime,
              },
            });
          }
        });
      } catch {
        // RaceEntry may not exist yet
      }

      // Sort all activities by date
      activities.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());

      // Apply limit
      const limited = activities.slice(0, limit);

      // Determine next cursor
      const hasMore = activities.length > limit;
      const nextCursor = hasMore && limited.length > 0
        ? limited[limited.length - 1].date
        : undefined;

      res.json({
        items: limited,
        nextCursor,
        hasMore,
      });
    } catch (error) {
      console.error('Error fetching activities:', error);
      res.status(500).json({ error: 'Failed to fetch activities' });
    }
  }
);

export default router;
```

```typescript
import { useInfiniteQuery } from '@tanstack/react-query';
import { useAuthStore } from '../../../../stores/authStore';
import type { ActivityFeedResponse } from '../../../types/activity';

export const activityKeys = {
  all: ['activities'] as const,
  feed: (athleteId?: string) => [...activityKeys.all, 'feed', athleteId] as const,
};

export function useActivityFeed(athleteId?: string) {
  const { accessToken, teamId } = useAuthStore();

  return useInfiniteQuery({
    queryKey: activityKeys.feed(athleteId),
    queryFn: async ({ pageParam }): Promise<ActivityFeedResponse> => {
      const params = new URLSearchParams();
      if (athleteId) params.set('athleteId', athleteId);
      if (pageParam) params.set('cursor', pageParam);
      params.set('limit', '20');

      const response = await fetch(`/api/v1/activities/unified?${params}`, {
        headers: {
          Authorization: `Bearer ${accessToken}`,
          'X-Team-ID': teamId || '',
        },
      });

      if (!response.ok) throw new Error('Failed to fetch activities');
      return response.json();
    },
    initialPageParam: undefined as string | undefined,
    getNextPageParam: (lastPage) => lastPage.nextCursor,
    enabled: !!accessToken && !!teamId,
    staleTime: 5 * 60 * 1000,
  });
}
```

```typescript
import { formatDistance } from 'date-fns';
import { Timer, Users, Trophy, Check, ArrowsLeftRight, Barbell } from '@phosphor-icons/react';
import type { AnyActivity, ActivityType } from '../../../types/activity';
import { ACTIVITY_TYPE_COLORS, ACTIVITY_TYPE_LABELS } from '../../../types/activity';

const TYPE_ICONS: Record<ActivityType, React.ElementType> = {
  erg_test: Timer,
  session_participation: Barbell,
  race_result: Trophy,
  attendance: Check,
  seat_race: ArrowsLeftRight,
  lineup_assignment: Users,
};

interface ActivityCardProps {
  activity: AnyActivity;
}

export function ActivityCard({ activity }: ActivityCardProps) {
  const Icon = TYPE_ICONS[activity.type];

  const formatErgTime = (seconds: number) => {
    const mins = Math.floor(seconds / 60);
    const secs = Math.floor(seconds % 60);
    const tenths = Math.floor((seconds % 1) * 10);
    return `${mins}:${secs.toString().padStart(2, '0')}.${tenths}`;
  };

  return (
    <div className="bg-surface-elevated rounded-lg border border-bdr-default p-4 hover:border-bdr-focus transition-colors">
      <div className="flex items-start gap-3">
        {/* Icon */}
        <div className={`w-10 h-10 rounded-full flex items-center justify-center shrink-0 ${ACTIVITY_TYPE_COLORS[activity.type]}`}>
          <Icon className="w-5 h-5" />
        </div>

        {/* Content */}
        <div className="flex-1 min-w-0">
          {/* Header */}
          <div className="flex items-center justify-between gap-2">
            <span className="font-medium text-txt-primary truncate">{activity.title}</span>
            <span className="text-xs text-txt-muted whitespace-nowrap">
              {formatDistance(new Date(activity.date), new Date(), { addSuffix: true })}
            </span>
          </div>

          {/* Description */}
          {activity.description && (
            <p className="text-sm text-txt-secondary mt-1">{activity.description}</p>
          )}

          {/* Type-specific details */}
          {activity.type === 'erg_test' && (
            <div className="flex items-center gap-4 mt-2 text-sm text-txt-secondary">
              <span>Time: {formatErgTime(activity.metadata.time)}</span>
              {activity.metadata.watts && <span>{activity.metadata.watts}W</span>}
              {activity.metadata.personalBest && (
                <span className="text-green-500 font-medium">PB!</span>
              )}
            </div>
          )}

          {activity.type === 'race_result' && (
            <div className="flex items-center gap-4 mt-2 text-sm text-txt-secondary">
              <span>Place: #{activity.metadata.place}</span>
              <span>{activity.metadata.boatClass}</span>
            </div>
          )}

          {activity.type === 'session_participation' && (
            <div className="flex items-center gap-4 mt-2 text-sm text-txt-secondary">
              <span>{activity.metadata.sessionType}</span>
              <span>{Math.round(activity.metadata.participationPercent)}% participation</span>
            </div>
          )}

          {/* Type badge */}
          <div className="mt-2">
            <span className={`inline-flex items-center px-2 py-0.5 rounded text-xs font-medium ${ACTIVITY_TYPE_COLORS[activity.type]}`}>
              {ACTIVITY_TYPE_LABELS[activity.type]}
            </span>
          </div>
        </div>
      </div>
    </div>
  );
}
```

```typescript
import { useEffect, useMemo } from 'react';
import { useInView } from 'react-intersection-observer';
import { isToday, isYesterday, startOfWeek } from 'date-fns';
import { useActivityFeed } from '../hooks/useActivityFeed';
import { ActivityCard } from './ActivityCard';
import type { AnyActivity } from '../../../types/activity';

interface ActivityFeedProps {
  athleteId?: string;
}

// Group activities by date
function groupByDate(activities: AnyActivity[]) {
  const groups: { label: string; activities: AnyActivity[] }[] = [];
  const weekStart = startOfWeek(new Date());

  const today: AnyActivity[] = [];
  const yesterday: AnyActivity[] = [];
  const thisWeek: AnyActivity[] = [];
  const earlier: AnyActivity[] = [];

  activities.forEach((activity) => {
    const date = new Date(activity.date);
    if (isToday(date)) {
      today.push(activity);
    } else if (isYesterday(date)) {
      yesterday.push(activity);
    } else if (date >= weekStart) {
      thisWeek.push(activity);
    } else {
      earlier.push(activity);
    }
  });

  if (today.length > 0) groups.push({ label: 'Today', activities: today });
  if (yesterday.length > 0) groups.push({ label: 'Yesterday', activities: yesterday });
  if (thisWeek.length > 0) groups.push({ label: 'This Week', activities: thisWeek });
  if (earlier.length > 0) groups.push({ label: 'Earlier', activities: earlier });

  return groups;
}

export function ActivityFeed({ athleteId }: ActivityFeedProps) {
  const { ref, inView } = useInView({
    threshold: 0.5,
    rootMargin: '100px',
  });

  const {
    data,
    isLoading,
    isError,
    fetchNextPage,
    hasNextPage,
    isFetchingNextPage,
  } = useActivityFeed(athleteId);

  // Auto-load when sentinel in view
  useEffect(() => {
    if (inView && hasNextPage && !isFetchingNextPage) {
      fetchNextPage();
    }
  }, [inView, hasNextPage, isFetchingNextPage, fetchNextPage]);

  // Flatten all pages
  const allActivities = useMemo(() => {
    return data?.pages.flatMap((page) => page.items) ?? [];
  }, [data]);

  // Group by date
  const groupedActivities = useMemo(() => {
    return groupByDate(allActivities);
  }, [allActivities]);

  if (isLoading) {
    return (
      <div className="space-y-4">
        {[...Array(5)].map((_, i) => (
          <div key={i} className="bg-surface-elevated rounded-lg border border-bdr-default p-4 animate-pulse">
            <div className="flex items-start gap-3">
              <div className="w-10 h-10 rounded-full bg-surface-default" />
              <div className="flex-1 space-y-2">
                <div className="h-4 bg-surface-default rounded w-3/4" />
                <div className="h-3 bg-surface-default rounded w-1/2" />
              </div>
            </div>
          </div>
        ))}
      </div>
    );
  }

  if (isError) {
    return (
      <div className="bg-red-500/10 border border-red-500/20 rounded-lg p-4 text-red-500 text-center">
        Failed to load activity feed. Please try again.
      </div>
    );
  }

  if (allActivities.length === 0) {
    return (
      <div className="text-center py-12 text-txt-muted">
        <p>No activities yet.</p>
      </div>
    );
  }

  return (
    <div className="space-y-6">
      {groupedActivities.map((group) => (
        <div key={group.label}>
          <h3 className="text-sm font-medium text-txt-muted mb-3">{group.label}</h3>
          <div className="space-y-3">
            {group.activities.map((activity) => (
              <ActivityCard key={activity.id} activity={activity} />
            ))}
          </div>
        </div>
      ))}

      {/* Infinite scroll sentinel */}
      <div ref={ref} className="py-4 text-center">
        {isFetchingNextPage ? (
          <div className="animate-spin w-6 h-6 border-2 border-accent-primary border-t-transparent rounded-full mx-auto" />
        ) : hasNextPage ? (
          <span className="text-txt-muted">Scroll for more</span>
        ) : (
          <span className="text-txt-muted">You've reached the end</span>
        )}
      </div>
    </div>
  );
}
```
