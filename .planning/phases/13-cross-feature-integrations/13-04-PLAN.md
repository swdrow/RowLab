```typescript
import Fuse from 'fuse.js';
import type { SearchResult, SearchResultType, SearchGroup } from '../../../types/search';

// Build search index from fetched data
export function buildSearchIndex(
  athletes: Array<{ id: string; firstName: string; lastName: string; sidePreference?: string }>,
  sessions: Array<{ id: string; name: string; type: string; date: string }>,
  ergTests: Array<{ id: string; athleteId: string; athleteName?: string; testType: string; date: string }>,
  lineups: Array<{ id: string; name: string; boatClass: string; date?: string }>,
  regattas: Array<{ id: string; name: string; location?: string; startDate: string }>,
  seatRaces: Array<{ id: string; date: string; conditions?: string }>
): SearchResult[] {
  const results: SearchResult[] = [];

  // Athletes
  athletes.forEach((a) => {
    results.push({
      id: a.id,
      type: 'athlete',
      title: `${a.firstName} ${a.lastName}`,
      subtitle: a.sidePreference ? `${a.sidePreference} side` : undefined,
      href: `/app/athletes/${a.id}`,
    });
  });

  // Sessions
  sessions.forEach((s) => {
    results.push({
      id: s.id,
      type: 'session',
      title: s.name,
      subtitle: `${s.type} - ${new Date(s.date).toLocaleDateString()}`,
      href: `/app/training/sessions/${s.id}`,
    });
  });

  // Erg Tests
  ergTests.forEach((e) => {
    results.push({
      id: e.id,
      type: 'erg_test',
      title: `${e.testType} Test`,
      subtitle: e.athleteName ? `${e.athleteName} - ${new Date(e.date).toLocaleDateString()}` : new Date(e.date).toLocaleDateString(),
      href: `/app/erg/${e.id}`,
    });
  });

  // Lineups
  lineups.forEach((l) => {
    results.push({
      id: l.id,
      type: 'lineup',
      title: l.name,
      subtitle: l.boatClass,
      href: `/app/lineups/${l.id}`,
    });
  });

  // Regattas
  regattas.forEach((r) => {
    results.push({
      id: r.id,
      type: 'regatta',
      title: r.name,
      subtitle: r.location || new Date(r.startDate).toLocaleDateString(),
      href: `/app/regattas/${r.id}`,
    });
  });

  // Seat Races
  seatRaces.forEach((s) => {
    results.push({
      id: s.id,
      type: 'seat_race',
      title: `Seat Race - ${new Date(s.date).toLocaleDateString()}`,
      subtitle: s.conditions,
      href: `/app/seat-racing/sessions/${s.id}`,
    });
  });

  return results;
}

// Search using Fuse.js for fuzzy matching
export function searchItems(
  items: SearchResult[],
  query: string
): SearchGroup[] {
  if (!query.trim()) return [];

  const fuse = new Fuse(items, {
    keys: ['title', 'subtitle'],
    threshold: 0.3,
    includeScore: true,
  });

  const results = fuse.search(query);

  // Group by type
  const grouped: Record<SearchResultType, SearchResult[]> = {
    athlete: [],
    session: [],
    erg_test: [],
    lineup: [],
    regatta: [],
    race: [],
    seat_race: [],
  };

  results.forEach(({ item }) => {
    grouped[item.type].push(item);
  });

  // Convert to SearchGroup array, filter empty groups
  const groups: SearchGroup[] = [];
  const typeOrder: SearchResultType[] = ['athlete', 'session', 'erg_test', 'lineup', 'regatta', 'race', 'seat_race'];

  typeOrder.forEach((type) => {
    if (grouped[type].length > 0) {
      groups.push({
        type,
        label: {
          athlete: 'Athletes',
          session: 'Sessions',
          erg_test: 'Erg Tests',
          lineup: 'Lineups',
          regatta: 'Regattas',
          race: 'Races',
          seat_race: 'Seat Races',
        }[type],
        results: grouped[type].slice(0, 5), // Limit per group
      });
    }
  });

  return groups;
}

// Local storage key for recent items
const RECENT_ITEMS_KEY = 'rowlab_recent_search_items';

// Get recent items from localStorage
export function getRecentItems(): SearchResult[] {
  try {
    const stored = localStorage.getItem(RECENT_ITEMS_KEY);
    return stored ? JSON.parse(stored) : [];
  } catch {
    return [];
  }
}

// Add item to recent items
export function addRecentItem(item: SearchResult): void {
  try {
    const recent = getRecentItems();
    // Remove if already exists
    const filtered = recent.filter((r) => r.id !== item.id || r.type !== item.type);
    // Add to front, limit to 10
    const updated = [item, ...filtered].slice(0, 10);
    localStorage.setItem(RECENT_ITEMS_KEY, JSON.stringify(updated));
  } catch {
    // Ignore localStorage errors
  }
}
```

Create useGlobalSearch.ts:
```typescript
import { useState, useEffect, useMemo, useCallback } from 'react';
import { useQuery } from '@tanstack/react-query';
import { useAuthStore } from '../../../../stores/authStore';
import { buildSearchIndex, searchItems, getRecentItems, addRecentItem } from '../services/searchService';
import type { SearchResult, SearchGroup } from '../../../types/search';

export function useGlobalSearch() {
  const [query, setQuery] = useState('');
  const [recentItems, setRecentItems] = useState<SearchResult[]>([]);
  const { accessToken, teamId } = useAuthStore();

  // Load recent items on mount
  useEffect(() => {
    setRecentItems(getRecentItems());
  }, []);

  // Fetch all searchable entities
  const { data: searchData, isLoading } = useQuery({
    queryKey: ['global-search-index', teamId],
    queryFn: async () => {
      const headers = {
        Authorization: `Bearer ${accessToken}`,
        'X-Team-ID': teamId || '',
      };

      // Fetch all entity types in parallel
      const [athletes, sessions, ergTests, lineups, regattas, seatRaces] = await Promise.all([
        fetch('/api/v1/athletes', { headers }).then((r) => r.json()).catch(() => []),
        fetch('/api/v1/sessions', { headers }).then((r) => r.json()).then((d) => d.sessions || []).catch(() => []),
        fetch('/api/v1/erg-tests', { headers }).then((r) => r.json()).catch(() => []),
        fetch('/api/v1/lineups', { headers }).then((r) => r.json()).catch(() => []),
        fetch('/api/v1/regattas', { headers }).then((r) => r.json()).then((d) => d.regattas || []).catch(() => []),
        fetch('/api/v1/seat-races/sessions', { headers }).then((r) => r.json()).catch(() => []),
      ]);

      return { athletes, sessions, ergTests, lineups, regattas, seatRaces };
    },
    enabled: !!accessToken && !!teamId,
    staleTime: 5 * 60 * 1000, // 5 minutes
  });

  // Build search index
  const searchIndex = useMemo(() => {
    if (!searchData) return [];
    return buildSearchIndex(
      searchData.athletes,
      searchData.sessions,
      searchData.ergTests,
      searchData.lineups,
      searchData.regattas,
      searchData.seatRaces
    );
  }, [searchData]);

  // Search results
  const results: SearchGroup[] = useMemo(() => {
    return searchItems(searchIndex, query);
  }, [searchIndex, query]);

  // Handle item selection
  const selectItem = useCallback((item: SearchResult) => {
    addRecentItem(item);
    setRecentItems(getRecentItems());
    return item.href;
  }, []);

  return {
    query,
    setQuery,
    isLoading,
    results,
    recentItems,
    selectItem,
  };
}
```

Create the cmdk-based command palette:

```typescript
import { useEffect, useState, useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import { Command } from 'cmdk';
import { motion, AnimatePresence } from 'framer-motion';
import {
  MagnifyingGlass,
  User,
  Calendar,
  ChartLine,
  Users,
  Trophy,
  Flag,
  ArrowsLeftRight,
  Clock,
} from '@phosphor-icons/react';
import { useGlobalSearch } from '../hooks/useGlobalSearch';
import type { SearchResult, SearchResultType } from '../../../types/search';

// Icon mapping
const TYPE_ICONS: Record<SearchResultType, React.ElementType> = {
  athlete: User,
  session: Calendar,
  erg_test: ChartLine,
  lineup: Users,
  regatta: Trophy,
  race: Flag,
  seat_race: ArrowsLeftRight,
};

interface CommandPaletteProps {
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export function CommandPalette({ open, onOpenChange }: CommandPaletteProps) {
  const navigate = useNavigate();
  const { query, setQuery, isLoading, results, recentItems, selectItem } = useGlobalSearch();

  // Handle keyboard shortcut
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'k' && (e.metaKey || e.ctrlKey)) {
        e.preventDefault();
        onOpenChange(!open);
      }
    };

    document.addEventListener('keydown', handleKeyDown);
    return () => document.removeEventListener('keydown', handleKeyDown);
  }, [open, onOpenChange]);

  // Handle item selection
  const handleSelect = useCallback(
    (item: SearchResult) => {
      const href = selectItem(item);
      onOpenChange(false);
      setQuery('');
      navigate(href);
    },
    [selectItem, onOpenChange, setQuery, navigate]
  );

  // Render search result item
  const renderItem = (item: SearchResult) => {
    const Icon = TYPE_ICONS[item.type];
    return (
      <Command.Item
        key={`${item.type}-${item.id}`}
        value={`${item.title} ${item.subtitle || ''}`}
        onSelect={() => handleSelect(item)}
        className="flex items-center gap-3 px-3 py-2 rounded-lg cursor-pointer
          text-txt-primary hover:bg-surface-hover
          data-[selected=true]:bg-accent-primary/10 data-[selected=true]:text-accent-primary"
      >
        <Icon className="w-5 h-5 text-txt-secondary shrink-0" />
        <div className="flex-1 min-w-0">
          <div className="truncate font-medium">{item.title}</div>
          {item.subtitle && (
            <div className="truncate text-sm text-txt-secondary">{item.subtitle}</div>
          )}
        </div>
      </Command.Item>
    );
  };

  return (
    <AnimatePresence>
      {open && (
        <>
          {/* Backdrop */}
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            className="fixed inset-0 bg-black/50 z-50"
            onClick={() => onOpenChange(false)}
          />

          {/* Dialog */}
          <motion.div
            initial={{ opacity: 0, scale: 0.95 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 0.95 }}
            transition={{ duration: 0.15 }}
            className="fixed top-[20%] left-1/2 -translate-x-1/2 w-full max-w-xl z-50"
          >
            <Command
              className="bg-surface-elevated rounded-xl shadow-2xl border border-bdr-default overflow-hidden"
              shouldFilter={false}
            >
              {/* Input */}
              <div className="flex items-center gap-3 px-4 py-3 border-b border-bdr-default">
                <MagnifyingGlass className="w-5 h-5 text-txt-secondary" />
                <Command.Input
                  value={query}
                  onValueChange={setQuery}
                  placeholder="Search athletes, sessions, erg tests..."
                  className="flex-1 bg-transparent outline-none text-txt-primary placeholder:text-txt-muted"
                  autoFocus
                />
                <kbd className="px-2 py-1 text-xs rounded bg-surface-default border border-bdr-default text-txt-secondary">
                  ESC
                </kbd>
              </div>

              {/* Results */}
              <Command.List className="max-h-[60vh] overflow-y-auto p-2">
                {isLoading && (
                  <Command.Loading>
                    <div className="px-3 py-6 text-center text-txt-secondary">
                      Loading...
                    </div>
                  </Command.Loading>
                )}

                <Command.Empty className="px-3 py-6 text-center text-txt-secondary">
                  No results found.
                </Command.Empty>

                {/* Recent items (shown when no query) */}
                {!query && recentItems.length > 0 && (
                  <Command.Group heading="Recent">
                    <div className="px-3 py-1.5 text-xs font-medium text-txt-muted flex items-center gap-2">
                      <Clock className="w-3.5 h-3.5" />
                      Recent
                    </div>
                    {recentItems.map(renderItem)}
                  </Command.Group>
                )}

                {/* Search results grouped by type */}
                {results.map((group) => (
                  <Command.Group key={group.type} heading={group.label}>
                    <div className="px-3 py-1.5 text-xs font-medium text-txt-muted">
                      {group.label}
                    </div>
                    {group.results.map(renderItem)}
                  </Command.Group>
                ))}
              </Command.List>

              {/* Footer */}
              <div className="flex items-center justify-between px-4 py-2 border-t border-bdr-default text-xs text-txt-secondary">
                <div className="flex items-center gap-4">
                  <span className="flex items-center gap-1">
                    <kbd className="px-1.5 py-0.5 rounded bg-surface-default border border-bdr-default">↑</kbd>
                    <kbd className="px-1.5 py-0.5 rounded bg-surface-default border border-bdr-default">↓</kbd>
                    to navigate
                  </span>
                  <span className="flex items-center gap-1">
                    <kbd className="px-1.5 py-0.5 rounded bg-surface-default border border-bdr-default">↵</kbd>
                    to select
                  </span>
                </div>
                <span className="flex items-center gap-1">
                  <kbd className="px-1.5 py-0.5 rounded bg-surface-default border border-bdr-default">⌘</kbd>
                  <kbd className="px-1.5 py-0.5 rounded bg-surface-default border border-bdr-default">K</kbd>
                  to toggle
                </span>
              </div>
            </Command>
          </motion.div>
        </>
      )}
    </AnimatePresence>
  );
}

// Export standalone hook for global registration
export function useCommandPalette() {
  const [open, setOpen] = useState(false);
  return { open, setOpen };
}
```
