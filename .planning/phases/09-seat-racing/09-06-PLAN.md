```tsx
function PieceAssignmentsReview({ piece, index }) {
```

**ValidationSummary sub-component:**
```tsx
function ValidationSummary({ data }) {
  const issues = [];

  // Check for missing times
  data.pieces.forEach((piece, i) => {
    piece.boats.forEach((boat, j) => {
      if (!boat.finishTimeSeconds) {
        issues.push(`Piece ${i+1}, ${boat.name}: Missing finish time`);
      }
    });
  });

  // Check for missing assignments
  data.pieces.forEach((piece, i) => {
    piece.boats.forEach((boat, j) => {
      if (!boat.assignments || boat.assignments.length === 0) {
        issues.push(`Piece ${i+1}, ${boat.name}: No athletes assigned`);
      }
    });
  });

  if (issues.length === 0) {
    return (
      <div className="p-3 bg-green-500/10 text-green-600 rounded-lg flex items-center gap-2">
        <CheckCircleIcon className="w-5 h-5" />
        Ready to submit! All data looks good.
      </div>
    );
  }

  return (
    <div className="p-3 bg-orange-500/10 text-orange-600 rounded-lg">
      <div className="flex items-center gap-2 mb-2">
        <AlertTriangleIcon className="w-5 h-5" />
        <span className="font-medium">Warnings (you can still submit)</span>
      </div>
      <ul className="list-disc list-inside text-sm">
        {issues.map((issue, i) => <li key={i}>{issue}</li>)}
      </ul>
    </div>
  );
}
```

**API submission strategy (CONFIRMED from server/routes/seatRaces.js):**
The API uses hierarchical multi-POST pattern (NOT single nested POST):
1. POST /api/v1/seat-races → Create session (returns session.id)
2. POST /api/v1/seat-races/:sessionId/pieces → Add each piece (returns piece.id)
3. POST /api/v1/seat-races/pieces/:pieceId/boats → Add each boat (returns boat.id)
4. PUT /api/v1/seat-races/boats/:boatId/assignments → Set assignments for each boat
5. POST /api/v1/seat-races/:sessionId/process → Calculate ratings

Use this hierarchical submission pattern:
```typescript
async function submitSession(data, mutations) {
  // Create session
  const sessionResult = await mutations.createSession.mutateAsync({
    date: new Date(data.date).toISOString(),
    boatClass: data.boatClass,
    conditions: data.conditions,
    location: data.location,
    description: data.description,
  });
  const sessionId = sessionResult.session.id;

  // Add pieces with boats and assignments
  for (const piece of data.pieces) {
    const pieceResult = await mutations.addPiece.mutateAsync({
      sessionId,
      sequenceOrder: piece.sequenceOrder,
      distanceMeters: piece.distanceMeters,
      direction: piece.direction,
      notes: piece.notes,
    });
    const pieceId = pieceResult.piece.id;

    for (const boat of piece.boats) {
      const boatResult = await mutations.addBoat.mutateAsync({
        pieceId,
        name: boat.name,
        finishTimeSeconds: boat.finishTimeSeconds,
        handicapSeconds: boat.handicapSeconds || 0,
      });
      const boatId = boatResult.boat.id;

      if (boat.assignments && boat.assignments.length > 0) {
        await mutations.setAssignments.mutateAsync({
          boatId,
          assignments: boat.assignments,
        });
      }
    }
  }

  // Process ratings
  await mutations.processSession.mutateAsync({ sessionId });

  return sessionId;
}
```
