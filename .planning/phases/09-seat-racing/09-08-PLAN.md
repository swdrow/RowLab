```javascript
import express from 'express';
import { authenticateToken, requireTeam } from '../middleware/auth.js';
import * as eloService from '../services/eloRatingService.js';
import logger from '../utils/logger.js';

const router = express.Router();

// All routes require authentication and team context
router.use(authenticateToken, requireTeam);

/**
 * GET /api/v1/ratings
 * Get athlete ratings for the active team
 * Query params:
 *   - type: Rating type (default: 'seat_race_elo')
 *   - minRaces: Minimum races to include (default: 0)
 *   - side: Filter by athlete side ('Port', 'Starboard')
 */
router.get('/', async (req, res) => {
  try {
    const { type = 'seat_race_elo', minRaces = '0', side } = req.query;

    const options = {
      ratingType: type,
      minRaces: parseInt(minRaces, 10),
    };

    let rankings = await eloService.getTeamRankings(req.user.activeTeamId, options);

    // Filter by side if specified
    if (side && (side === 'Port' || side === 'Starboard')) {
      rankings = rankings.filter(r => {
        const athleteSide = r.athlete?.side;
        return athleteSide === side || athleteSide === 'Both';
      });
    }

    res.json({ success: true, data: { ratings: rankings } });
  } catch (err) {
    logger.error('Get ratings error', { error: err.message });
    res.status(500).json({
      success: false,
      error: { code: 'SERVER_ERROR', message: 'Failed to fetch ratings' },
    });
  }
});

/**
 * GET /api/v1/ratings/:athleteId
 * Get rating for a specific athlete
 */
router.get('/:athleteId', async (req, res) => {
  try {
    const { athleteId } = req.params;
    const { type = 'seat_race_elo' } = req.query;

    const rating = await eloService.getOrCreateRating(
      athleteId,
      req.user.activeTeamId,
      type
    );

    res.json({ success: true, data: { rating } });
  } catch (err) {
    logger.error('Get athlete rating error', { error: err.message });
    res.status(500).json({
      success: false,
      error: { code: 'SERVER_ERROR', message: 'Failed to fetch athlete rating' },
    });
  }
});

/**
 * POST /api/v1/ratings/recalculate
 * Recalculate all ratings from scratch (OWNER, COACH only)
 */
router.post('/recalculate', async (req, res) => {
  try {
    // Check role
    const membership = req.user.memberships?.find(
      m => m.teamId === req.user.activeTeamId
    );
    if (!membership || !['OWNER', 'COACH'].includes(membership.role)) {
      return res.status(403).json({
        success: false,
        error: { code: 'FORBIDDEN', message: 'Insufficient permissions' },
      });
    }

    const { type = 'seat_race_elo' } = req.body;

    const result = await eloService.recalculateAllRatings(
      req.user.activeTeamId,
      type
    );

    res.json({ success: true, data: result });
  } catch (err) {
    logger.error('Recalculate ratings error', { error: err.message });
    res.status(500).json({
      success: false,
      error: { code: 'SERVER_ERROR', message: 'Failed to recalculate ratings' },
    });
  }
});

/**
 * GET /api/v1/ratings/parameters
 * Get current rating parameters (K-factor, etc)
 */
router.get('/parameters', async (req, res) => {
  try {
    // For now, return static defaults
    // Future: could be stored per-team in database
    res.json({
      success: true,
      data: {
        parameters: {
          kFactor: eloService.K_FACTOR,
          defaultRating: eloService.DEFAULT_RATING,
          drawThreshold: 0.5, // seconds
          marginScaling: true,
          maxMarginFactor: 2,
        },
      },
    });
  } catch (err) {
    logger.error('Get parameters error', { error: err.message });
    res.status(500).json({
      success: false,
      error: { code: 'SERVER_ERROR', message: 'Failed to fetch parameters' },
    });
  }
});

export default router;
```

```tsx
import { useQuery } from '@tanstack/react-query';
import { InfoIcon, SettingsIcon } from 'lucide-react';

interface RatingParameters {
  kFactor: number;
  defaultRating: number;
  drawThreshold: number;
  marginScaling: boolean;
  maxMarginFactor: number;
}

function useRatingParameters() {
  return useQuery({
    queryKey: ['ratingParameters'],
    queryFn: async () => {
      const res = await fetch('/api/v1/ratings/parameters');
      const json = await res.json();
      if (!json.success) throw new Error(json.error?.message);
      return json.data.parameters as RatingParameters;
    },
    staleTime: 60 * 60 * 1000, // 1 hour - parameters rarely change
  });
}

interface ParametersPanelProps {
  className?: string;
}

export function ParametersPanel({ className = '' }: ParametersPanelProps) {
  const { data: params, isLoading } = useRatingParameters();

  if (isLoading) {
    return (
      <div className={`bg-surface rounded-lg p-4 ${className}`}>
        <div className="animate-pulse space-y-3">
          <div className="h-5 bg-surface-elevated rounded w-1/3" />
          <div className="h-4 bg-surface-elevated rounded w-2/3" />
          <div className="h-4 bg-surface-elevated rounded w-1/2" />
        </div>
      </div>
    );
  }

  if (!params) return null;

  return (
    <div className={`bg-surface rounded-lg p-4 ${className}`}>
      <div className="flex items-center gap-2 mb-4">
        <SettingsIcon className="w-4 h-4 text-muted" />
        <h3 className="font-medium">Rating Parameters</h3>
      </div>

      <div className="space-y-3 text-sm">
        <ParameterRow
          label="K-Factor"
          value={params.kFactor}
          description="Rating volatility (higher = more change per race)"
        />
        <ParameterRow
          label="Default Rating"
          value={params.defaultRating}
          description="Starting ELO for new athletes"
        />
        <ParameterRow
          label="Draw Threshold"
          value={`${params.drawThreshold}s`}
          description="Time difference for a draw"
        />
        <ParameterRow
          label="Margin Scaling"
          value={params.marginScaling ? 'Enabled' : 'Disabled'}
          description="Larger margins increase rating change"
        />
        {params.marginScaling && (
          <ParameterRow
            label="Max Margin Factor"
            value={`${params.maxMarginFactor}x`}
            description="Maximum K-factor multiplier for large margins"
          />
        )}
      </div>

      <div className="mt-4 p-3 bg-surface-elevated rounded text-xs text-muted">
        <InfoIcon className="w-3 h-3 inline mr-1" />
        Parameters are currently fixed. Contact support to adjust team settings.
      </div>
    </div>
  );
}

interface ParameterRowProps {
  label: string;
  value: string | number;
  description: string;
}

function ParameterRow({ label, value, description }: ParameterRowProps) {
  return (
    <div className="flex justify-between items-start">
      <div>
        <div className="font-medium text-primary">{label}</div>
        <div className="text-xs text-muted">{description}</div>
      </div>
      <div className="font-mono text-primary">{value}</div>
    </div>
  );
}
```
