```typescript
import { useMemo, useState } from 'react';
import type { Athlete } from '@v2/types/athletes';

interface AthleteAvatarProps {
  athlete: Pick<Athlete, 'firstName' | 'lastName'> & { photoUrl?: string };
  size?: 'sm' | 'md' | 'lg';
  className?: string;
}

const sizeClasses = {
  sm: 'w-8 h-8 text-xs',
  md: 'w-12 h-12 text-sm',
  lg: 'w-16 h-16 text-base',
};

const bgColors = [
  'bg-blue-500',
  'bg-green-500',
  'bg-purple-500',
  'bg-orange-500',
  'bg-pink-500',
  'bg-teal-500',
  'bg-indigo-500',
  'bg-amber-500',
];

export function AthleteAvatar({ athlete, size = 'md', className = '' }: AthleteAvatarProps) {
  const [imgError, setImgError] = useState(false);
  const initials = `${athlete.firstName[0] || ''}${athlete.lastName[0] || ''}`.toUpperCase();

  // Deterministic background color from name
  const bgColor = useMemo(() => {
    const hash = (athlete.firstName.charCodeAt(0) || 0) + (athlete.lastName.charCodeAt(0) || 0);
    return bgColors[hash % bgColors.length];
  }, [athlete.firstName, athlete.lastName]);

  const photoUrl = 'photoUrl' in athlete ? athlete.photoUrl : undefined;

  return (
    <div className={`rounded-full overflow-hidden flex-shrink-0 ${sizeClasses[size]} ${className}`}>
      {photoUrl && !imgError ? (
        <img
          src={photoUrl}
          alt={`${athlete.firstName} ${athlete.lastName}`}
          onError={() => setImgError(true)}
          className="w-full h-full object-cover"
        />
      ) : (
        <div
          className={`${bgColor} w-full h-full flex items-center justify-center text-white font-semibold`}
        >
          {initials}
        </div>
      )}
    </div>
  );
}
```

```typescript
// src/v2/components/athletes/AthleteFilters.tsx
import { Search, X } from 'lucide-react';
import type { AthleteFilters as Filters, SidePreference } from '@v2/types/athletes';

interface AthleteFiltersProps {
  filters: Filters;
  onChange: (filters: Filters) => void;
}

export function AthleteFilters({ filters, onChange }: AthleteFiltersProps) {
  const updateFilter = <K extends keyof Filters>(key: K, value: Filters[K]) => {
    onChange({ ...filters, [key]: value });
  };

  const clearFilters = () => {
    onChange({ search: '', side: 'all', canScull: null, canCox: null });
  };

  const hasActiveFilters =
    filters.search ||
    (filters.side && filters.side !== 'all') ||
    filters.canScull !== null ||
    filters.canCox !== null;

  return (
    <div className="flex flex-wrap items-center gap-3">
      {/* Search input */}
      <div className="relative">
        <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-txt-secondary" />
        <input
          type="text"
          placeholder="Search athletes..."
          value={filters.search || ''}
          onChange={(e) => updateFilter('search', e.target.value)}
          className="pl-10 pr-4 py-2 w-64 bg-bg-surface-elevated border border-bdr-default rounded-lg
                     text-txt-primary placeholder-txt-secondary focus:outline-none focus:border-interactive-primary"
        />
      </div>

      {/* Side filter */}
      <select
        value={filters.side || 'all'}
        onChange={(e) => updateFilter('side', e.target.value as SidePreference | 'all')}
        className="px-3 py-2 bg-bg-surface-elevated border border-bdr-default rounded-lg
                   text-txt-primary focus:outline-none focus:border-interactive-primary"
      >
        <option value="all">All Sides</option>
        <option value="Port">Port</option>
        <option value="Starboard">Starboard</option>
        <option value="Both">Both</option>
        <option value="Cox">Cox</option>

      {/* Capability filters */}
      <div className="flex items-center gap-2">
        <label className="flex items-center gap-2 cursor-pointer">

          <span className="text-sm text-txt-secondary">Can Scull</span>

        >
          <X size={14} />
          Clear
        </button>
      )}
    </div>
  );
}
```

```typescript
// src/v2/components/athletes/AthletesTable.tsx
import { useMemo } from 'react';
import { VirtualTable } from '@v2/components/common';
import { AthleteAvatar } from './AthleteAvatar';
import type { ColumnDef } from '@tanstack/react-table';
import type { Athlete } from '@v2/types/athletes';

interface AthletesTableProps {
  athletes: Athlete[];
  isLoading: boolean;
  selectedId?: string;
  onSelect: (athlete: Athlete) => void;
}

export function AthletesTable({ athletes, isLoading, selectedId, onSelect }: AthletesTableProps) {
  const columns = useMemo<ColumnDef<Athlete>[]>(
    () => [
      {
        id: 'avatar',
        header: '',
        size: 60,
        enableSorting: false,
        cell: ({ row }) => <AthleteAvatar athlete={row.original} size="sm" />,
      },
      {
        accessorKey: 'lastName',
        header: 'Name',
        size: 200,
        cell: ({ row }) => (
          <span className="font-medium">
            {row.original.lastName}, {row.original.firstName}
          </span>
        ),
      },
      {
        accessorKey: 'side',
        header: 'Side',
        size: 100,
        cell: ({ getValue }) => {
          const side = getValue() as string | null;
          if (!side) return <span className="text-txt-tertiary">—</span>;
          const colors: Record<string, string> = {
            Port: 'text-red-400',
            Starboard: 'text-green-400',
            Both: 'text-blue-400',
            Cox: 'text-purple-400',
          };
          return <span className={colors[side] || ''}>{side}</span>;
        },
      },
      {
        accessorKey: 'heightCm',
        header: 'Height',
        size: 80,
        cell: ({ getValue }) => {
          const height = getValue() as number | null;
          return height ? `${height} cm` : <span className="text-txt-tertiary">—</span>;
        },
      },
      {
        accessorKey: 'weightKg',
        header: 'Weight',
        size: 80,
        cell: ({ getValue }) => {
          const weight = getValue() as number | null;
          return weight ? `${weight} kg` : <span className="text-txt-tertiary">—</span>;
        },
      },
      {
        id: 'capabilities',
        header: 'Capabilities',
        size: 120,
        enableSorting: false,
        cell: ({ row }) => {
          const caps = [];
          if (row.original.canScull) caps.push('Scull');
          if (row.original.canCox) caps.push('Cox');
          return caps.length > 0 ? (
            <span className="text-txt-secondary text-sm">{caps.join(', ')}</span>
          ) : (
            <span className="text-txt-tertiary">—</span>
          );
        },
      },
    ],
    []
  );

  return (
    <VirtualTable
      data={athletes}
      columns={columns}
      isLoading={isLoading}
      selectedId={selectedId}
      onRowClick={onSelect}
      emptyMessage="No athletes found. Add athletes or adjust your filters."
      className="h-[calc(100vh-280px)]"
    />
  );
}
```

```typescript
import { useEffect } from 'react';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { motion, AnimatePresence } from 'framer-motion';
import { X } from 'lucide-react';
import { AthleteAvatar } from './AthleteAvatar';
import type { Athlete, SidePreference } from '@v2/types/athletes';

const athleteSchema = z.object({
  firstName: z.string().min(1, 'First name is required'),
  lastName: z.string().min(1, 'Last name is required'),
  email: z.string().email().nullable().or(z.literal('')),
  side: z.enum(['Port', 'Starboard', 'Both', 'Cox']).nullable(),
  canScull: z.boolean(),
  canCox: z.boolean(),
  heightCm: z.number().min(100).max(250).nullable(),
  weightKg: z.number().min(30).max(200).nullable(),
});

type AthleteFormData = z.infer<typeof athleteSchema>;

interface AthleteEditPanelProps {
  athlete: Athlete | null;
  isOpen: boolean;
  onClose: () => void;
  onSave: (data: Partial<Athlete> & { id: string }) => void;
  isSaving: boolean;
}

export function AthleteEditPanel({
  athlete,
  isOpen,
  onClose,
  onSave,
  isSaving,
}: AthleteEditPanelProps) {
  const {
    register,
    handleSubmit,
    reset,
    formState: { errors, isDirty },
  } = useForm<AthleteFormData>({
    resolver: zodResolver(athleteSchema),
    defaultValues: {
      firstName: '',
      lastName: '',
      email: null,
      side: null,
      canScull: false,
      canCox: false,
      heightCm: null,
      weightKg: null,
    },
  });

  // Reset form when athlete changes
  useEffect(() => {
    if (athlete) {
      reset({
        firstName: athlete.firstName,
        lastName: athlete.lastName,
        email: athlete.email || '',
        side: athlete.side as SidePreference,
        canScull: athlete.canScull,
        canCox: athlete.canCox,
        heightCm: athlete.heightCm,
        weightKg: athlete.weightKg ? Number(athlete.weightKg) : null,
      });
    }
  }, [athlete, reset]);

  const onSubmit = (data: AthleteFormData) => {
    if (!athlete) return;
    onSave({
      id: athlete.id,
      ...data,
      email: data.email || null,
    });
  };

  // Handle escape key
  useEffect(() => {
    const handleEscape = (e: KeyboardEvent) => {
      if (e.key === 'Escape' && isOpen) {
        onClose();
      }
    };
    document.addEventListener('keydown', handleEscape);
    return () => document.removeEventListener('keydown', handleEscape);
  }, [isOpen, onClose]);

  return (
    <AnimatePresence>
      {isOpen && athlete && (
        <>
          {/* Backdrop */}
          <motion.div
            className="fixed inset-0 bg-black/50 z-40"
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            onClick={onClose}
          />

          {/* Panel */}
          <motion.div
            className="fixed right-0 top-0 h-full w-[400px] bg-bg-surface shadow-2xl z-50 overflow-y-auto"
            initial={{ x: '100%' }}
            animate={{ x: 0 }}
            exit={{ x: '100%' }}
            transition={{ type: 'spring', damping: 30, stiffness: 300 }}
          >
            {/* Header */}
            <div className="sticky top-0 bg-bg-surface border-b border-bdr-default px-6 py-4 flex items-center justify-between">
              <h2 className="text-lg font-semibold text-txt-primary">Edit Athlete</h2>
              <button
                onClick={onClose}
                className="p-2 text-txt-secondary hover:text-txt-primary hover:bg-bg-hover rounded-lg"
              >
                <X size={20} />
              </button>
            </div>

            {/* Avatar Section */}
            <div className="px-6 py-6 border-b border-bdr-default flex items-center gap-4">
              <AthleteAvatar athlete={athlete} size="lg" />
              <div>
                <h3 className="font-semibold text-txt-primary">
                  {athlete.firstName} {athlete.lastName}
                </h3>
                <p className="text-sm text-txt-secondary">
                  {athlete.email || 'No email'}
                </p>
              </div>
            </div>

            {/* Form */}
            <form onSubmit={handleSubmit(onSubmit)} className="px-6 py-6 space-y-5">
              {/* Name fields */}
              <div className="grid grid-cols-2 gap-4">
                <div>
                  <label className="block text-sm font-medium text-txt-secondary mb-1">
                    First Name
                  </label>
                  <input
                    {...register('firstName')}
                    className="w-full px-3 py-2 bg-bg-surface-elevated border border-bdr-default rounded-lg
                               text-txt-primary focus:outline-none focus:border-interactive-primary"
                  />
                  {errors.firstName && (
                    <p className="mt-1 text-xs text-status-error">{errors.firstName.message}</p>
                  )}
                </div>
                <div>
                  <label className="block text-sm font-medium text-txt-secondary mb-1">
                    Last Name
                  </label>
                  <input
                    {...register('lastName')}
                    className="w-full px-3 py-2 bg-bg-surface-elevated border border-bdr-default rounded-lg
                               text-txt-primary focus:outline-none focus:border-interactive-primary"
                  />
                  {errors.lastName && (
                    <p className="mt-1 text-xs text-status-error">{errors.lastName.message}</p>
                  )}
                </div>
              </div>

              {/* Email */}
              <div>
                <label className="block text-sm font-medium text-txt-secondary mb-1">
                  Email
                </label>
                <input
                  {...register('email')}
                  type="email"
                  className="w-full px-3 py-2 bg-bg-surface-elevated border border-bdr-default rounded-lg
                             text-txt-primary focus:outline-none focus:border-interactive-primary"
```

```typescript
// src/v2/pages/AthletesPage.tsx
import { useState, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { Plus, Upload } from 'lucide-react';
import { useAthletes } from '@v2/hooks/useAthletes';
import {
  AthletesTable,
  AthleteCard,
  ViewToggle,
  AthleteFilters,
  AthleteEditPanel,
} from '@v2/components/athletes';
import type { Athlete, AthleteFilters as Filters } from '@v2/types/athletes';

type ViewMode = 'grid' | 'list';

export default function AthletesPage() {
  // Filter state lifted above view toggle (per research Pitfall 4)
  const [filters, setFilters] = useState<Filters>({
    search: '',
    side: 'all',
    canScull: null,
    canCox: null,
  });

  // View preference (TODO: persist to user preferences API)
  const [view, setView] = useState<ViewMode>('list');

  // Selected athlete for editing
  const [selectedAthlete, setSelectedAthlete] = useState<Athlete | null>(null);
  const [isPanelOpen, setIsPanelOpen] = useState(false);

  // Fetch athletes with filters
  const {
    athletes,
    isLoading,
    updateAthlete,
    isUpdating,
  } = useAthletes(filters);

  const handleSelectAthlete = useCallback((athlete: Athlete) => {
    setSelectedAthlete(athlete);
    setIsPanelOpen(true);
  }, []);

  const handleClosePanel = useCallback(() => {
    setIsPanelOpen(false);
    // Delay clearing selection for exit animation
    setTimeout(() => setSelectedAthlete(null), 300);
  }, []);

  const handleSaveAthlete = useCallback(
    (data: Partial<Athlete> & { id: string }) => {
      updateAthlete(data, {
        onSuccess: () => {
          handleClosePanel();
        },
      });
    },
    [updateAthlete, handleClosePanel]
  );

  return (
    <div className="h-full">
      {/* Header */}
      <div className="px-6 py-4 border-b border-bdr-default">
        <div className="flex items-center justify-between mb-4">
          <div>
            <h1 className="text-2xl font-display font-semibold text-txt-primary">
              Athletes
            </h1>
            <p className="text-sm text-txt-secondary mt-1">
              {athletes.length} athlete{athletes.length !== 1 ? 's' : ''} in roster
            </p>
          </div>
          <div className="flex items-center gap-3">
            <button
              className="flex items-center gap-2 px-4 py-2 bg-bg-surface-elevated text-txt-primary
                         rounded-lg font-medium hover:bg-bg-hover border border-bdr-default"
            >
              <Upload size={18} />
              Import CSV
            </button>
            <button
              className="flex items-center gap-2 px-4 py-2 bg-interactive-primary text-white
                         rounded-lg font-medium hover:bg-interactive-hover"
            >
              <Plus size={18} />
              Add Athlete
            </button>
          </div>
        </div>

        {/* Filters and view toggle */}
        <div className="flex items-center justify-between">
          <AthleteFilters filters={filters} onChange={setFilters} />
          <ViewToggle view={view} onChange={setView} />
        </div>
      </div>

      {/* Content */}
      <div className="px-6 py-4">
        <AnimatePresence mode="wait">
          <motion.div
            key={view}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            exit={{ opacity: 0, y: -10 }}
            transition={{ duration: 0.15 }}
          >
            {view === 'list' ? (
              <AthletesTable
                athletes={athletes}
                isLoading={isLoading}
                selectedId={selectedAthlete?.id}
                onSelect={handleSelectAthlete}
              />
            ) : (
              <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
                {isLoading ? (
                  // Loading skeleton
                  Array.from({ length: 8 }).map((_, i) => (
                    <div
                      key={i}
                      className="h-24 bg-bg-surface-elevated rounded-xl animate-pulse"
                    />
                  ))
                ) : athletes.length === 0 ? (
                  <div className="col-span-full text-center py-12 text-txt-secondary">
                    No athletes found. Add athletes or adjust your filters.
                  </div>
                ) : (
                  athletes.map((athlete) => (
                    <AthleteCard
                      key={athlete.id}
                      athlete={athlete}
                      isSelected={selectedAthlete?.id === athlete.id}
                      onClick={() => handleSelectAthlete(athlete)}
                    />
                  ))
                )}
              </div>
            )}
          </motion.div>
        </AnimatePresence>
      </div>

      {/* Edit panel */}
      <AthleteEditPanel
        athlete={selectedAthlete}
        isOpen={isPanelOpen}
        onClose={handleClosePanel}
        onSave={handleSaveAthlete}
        isSaving={isUpdating}
      />
    </div>
  );
}
```
