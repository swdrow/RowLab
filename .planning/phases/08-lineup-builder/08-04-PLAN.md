3. Also support Cmd+Z on Mac (metaKey)
4. Prevent default browser behavior for these shortcuts
5. Only active when lineup builder is mounted (cleanup on unmount)

Implementation:
```typescript
import { useEffect } from 'react';
import useLineupStore from '../../store/lineupStore';

export function useLineupKeyboard() {
  const undo = useLineupStore(state => state.undo);
  const redo = useLineupStore(state => state.redo);

  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      // Check for Ctrl/Cmd + Z
      if ((e.ctrlKey || e.metaKey) && e.key === 'z') {
        e.preventDefault();
        if (e.shiftKey) {
          redo();
        } else {
          undo();
        }
      }
      // Also support Ctrl/Cmd + Y for redo (Windows standard)
      if ((e.ctrlKey || e.metaKey) && e.key === 'y') {
        e.preventDefault();
        redo();
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [undo, redo]);
}
```
